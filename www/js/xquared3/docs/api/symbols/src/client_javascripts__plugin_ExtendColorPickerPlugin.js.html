255 		this.scheme = null;

261 		this.parse = function(uri) {
262 			var m = uri.match(/^(([A-Za-z][0-9A-Za-z+.-]*)(:))?((\/\/)([^\/?#]*))?([^?#]*)((\?)([^#]*))?((#)(.*))?/);
263 			this.scheme = m[3] ? m[2] : null;
264 			this.authority = m[5] ? m[6] : null;
265 			this.path = m[7];
266 			this.query = m[9] ? m[10] : null;
267 			this.fragment = m[12] ? m[13] : null;
268 			return this;
269 		}

271 		this.toString = function() {
272 			var result = '';
273 			if(this.scheme != null) result = result + this.scheme + ':';
274 			if(this.authority != null) result = result + '//' + this.authority;
275 			if(this.path != null) result = result + this.path;
276 			if(this.query != null) result = result + '?' + this.query;
277 			if(this.fragment != null) result = result + '#' + this.fragment;
278 			return result;

281 		this.toAbsolute = function(base) {
282 			var base = new jscolor.URI(base);
283 			var r = this;
284 			var t = new jscolor.URI;
285 
286 			if(base.scheme == null) return false;
287 
288 			if(r.scheme != null && r.scheme.toLowerCase() == base.scheme.toLowerCase()) {
289 				r.scheme = null;
290 			}
291 
292 			if(r.scheme != null) {
293 				t.scheme = r.scheme;
294 				t.authority = r.authority;
295 				t.path = removeDotSegments(r.path);
296 				t.query = r.query;
297 			} else {
298 				if(r.authority != null) {
299 					t.authority = r.authority;
300 					t.path = removeDotSegments(r.path);
301 					t.query = r.query;
302 				} else {
303 					if(r.path == '') {
304 						t.path = base.path;
305 						if(r.query != null) {
306 							t.query = r.query;
307 						} else {
308 							t.query = base.query;
309 						}
310 					} else {
311 						if(r.path.substr(0,1) == '/') {
312 							t.path = removeDotSegments(r.path);
313 						} else {
314 							if(base.authority != null && base.path == '') {
315 								t.path = '/'+r.path;
316 							} else {
317 								t.path = base.path.replace(/[^\/]+$/,'')+r.path;
318 							}
319 							t.path = removeDotSegments(t.path);
320 						}
321 						t.query = r.query;
322 					}
323 					t.authority = base.authority;
324 				}
325 				t.scheme = base.scheme;
326 			}
327 			t.fragment = r.fragment;
328 
329 			return t;
330 		}

332 		function removeDotSegments(path) {
333 			var out = '';
334 			while(path) {
335 				if(path.substr(0,3)=='../' || path.substr(0,2)=='./') {
336 					path = path.replace(/^\.+/,'').substr(1);
337 				} else if(path.substr(0,3)=='/./' || path=='/.') {
338 					path = '/'+path.substr(3);
339 				} else if(path.substr(0,4)=='/../' || path=='/..') {
340 					path = '/'+path.substr(4);
341 					out = out.replace(/\/?[^\/]*$/, '');
342 				} else if(path=='.' || path=='..') {
343 					path = '';
344 				} else {
345 					var rm = path.match(/^\/?[^\/]*/)[0];
346 					path = path.substr(rm.length);
347 					out = out + rm;
348 				}
349 			}
350 			return out;
351 		}

515 		this.fromString = function(hex, flags) {
516 			var m = hex.match(/^\W*([0-9A-F]{3}([0-9A-F]{3})?)\W*$/i);
517 			if(!m) {
518 				return false;
519 			} else {
520 				if(m[1].length == 6) { // 6-char notation;
521 					this.fromRGB(
522 						parseInt(m[1].substr(0,2),16) / 255,
523 						parseInt(m[1].substr(2,2),16) / 255,
524 						parseInt(m[1].substr(4,2),16) / 255,
525 						flags
526 					);
527 				} else { // 3-char notation;
528 					this.fromRGB(
529 						parseInt(m[1].charAt(0)+m[1].charAt(0),16) / 255,
530 						parseInt(m[1].charAt(1)+m[1].charAt(1),16) / 255,
531 						parseInt(m[1].charAt(2)+m[1].charAt(2),16) / 255,
532 						flags
533 					);
534 				}
535 				return true;
536 			}

549 		function RGB_HSV(r, g, b) {
550 			var n = Math.min(Math.min(r,g),b);
551 			var v = Math.max(Math.max(r,g),b);
552 			var m = v - n;
553 			if(m == 0) return [ null, 0, v ];
554 			var h = r==n ? 3+(b-g)/m : (g==n ? 5+(r-b)/m : 1+(g-r)/m);
555 			return [ h==6?0:h, m/v, v ];
556 		}

559 		function HSV_RGB(h, s, v) {
560 			if(h == null) return [ v, v, v ];
561 			var i = Math.floor(h);
562 			var f = i%2 ? h-i : 1-(h-i);
563 			var m = v * (1 - s);
564 			var n = v * (1 - s*f);
565 			switch(i) {
566 				case 6:;
567 				case 0: return [v,n,m];
568 				case 1: return [n,v,m];
569 				case 2: return [m,v,n];
570 				case 3: return [m,n,v];
571 				case 4: return [n,m,v];
572 				case 5: return [v,m,n];
573 			}
574 		}

741 		function redrawPad() {
742 			// redraw the pad pointer;
743 			switch(modeID) {
744 				case 0: var yComponent = 1; break;
745 				case 1: var yComponent = 2; break;
746 			}
747 
748 			var x = Math.round((THIS.hsv[0]/6) * (jscolor.images.pad[0]-1));
749 			var y = Math.round((1-THIS.hsv[yComponent]) * (jscolor.images.pad[1]-1));
750 			jscolor.picker.padM.style.backgroundPosition =
751 				(THIS.pickerFace+THIS.pickerInset+x - Math.floor(jscolor.images.cross[0]/2)) + 'px ' +
752 				(THIS.pickerFace+THIS.pickerInset+y - Math.floor(jscolor.images.cross[1]/2)) + 'px';
753 			
754 			// redraw the slider image;
755 			var seg = jscolor.picker.sld.childNodes;
756 			
757 			switch(modeID) {
758 				case 0:;
759 					var rgb = HSV_RGB(THIS.hsv[0], THIS.hsv[1], 1);
760 					for(var i=0; i<seg.length; i++) {
761 						seg[i].style.backgroundColor = 'rgb('+
762 							(rgb[0]*(1-i/seg.length)*100)+'%,'+
763 							(rgb[1]*(1-i/seg.length)*100)+'%,'+
764 							(rgb[2]*(1-i/seg.length)*100)+'%)';
765 					}
766 					break;
767 				case 1:;
768 					var rgb, s, c = [ THIS.hsv[2], 0, 0 ];
769 					var i = Math.floor(THIS.hsv[0]);
770 					var f = i%2 ? THIS.hsv[0]-i : 1-(THIS.hsv[0]-i);
771 					switch(i) {
772 						case 6:;
773 						case 0: rgb=[0,1,2]; break;
774 						case 1: rgb=[1,0,2]; break;
775 						case 2: rgb=[2,0,1]; break;
776 						case 3: rgb=[2,1,0]; break;
777 						case 4: rgb=[1,2,0]; break;
778 						case 5: rgb=[0,2,1]; break;
779 					}
780 					for(var i=0; i<seg.length; i++) {
781 						s = 1 - 1/(seg.length-1)*i;
782 						c[1] = c[0] * (1 - s*f);
783 						c[2] = c[0] * (1 - s);
784 						seg[i].style.backgroundColor = 'rgb('+
785 							(c[rgb[0]]*100)+'%,'+
786 							(c[rgb[1]]*100)+'%,'+
787 							(c[rgb[2]]*100)+'%)';
788 					}
789 					break;
790 			}
791 		}
