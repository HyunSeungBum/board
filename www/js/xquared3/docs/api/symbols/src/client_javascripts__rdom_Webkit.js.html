1 /**

  3  * @requires rdom/W3.js
  4  */
  5 xq.rdom.Webkit = xq.Class(xq.rdom.W3,
  6 	/**
  7 	 * @name xq.rdom.Webkit
  8 	 * @lends xq.rdom.Webkit.prototype
  9 	 * @extends xq.rdom.Base
 10 	 * @constructor
 11 	 */
 12 	{
 13 	makePlaceHolder: function() {
 14 		var holder = this.createElement("BR");
 15 		holder.className = "webkit-block-placeholder";
 16 		return holder;
 17 	},
 18 	
 19 	makePlaceHolderString: function() {
 20 		return '<br class="webkit-block-placeholder" />';
 21 	},
 22 	
 23 	makeEmptyParagraph: function() {
 24 		return this.createElementFromHtml('<p><br class="webkit-block-placeholder" /></p>');
 25 	},
 26 	
 27 	isPlaceHolder: function(node) {
 28 		return node.className === "webkit-block-placeholder";
 29 	},
 30 
 31 	selectElement: function(element, entireElement) {
 32 		if(!element) throw "[element] is null";
 33 		if(element.nodeType !== 1) throw "[element] is not an element";
 34 		
 35 		var rng = this.rng() || this.getDoc().createRange();
 36 		if(entireElement) {
 37 			rng.selectNode(element);
 38 		} else {
 39 			rng.selectNodeContents(element);
 40 		}
 41 		
 42 		this._setSelectionByRange(rng);
 43 	},
 44 
 45 	getSelectionAsHtml: function() {
 46 		var container = this.createElement("div");
 47 		var rng = this.rng();
 48 		var contents = this.rng().cloneContents();
 49 		if(contents) container.appendChild(contents);
 50 		return container.innerHTML;
 51 	},
 52 
 53 	collapseSelection: function(toStart) {
 54 		var rng = this.rng();
 55 		rng.collapse(toStart);
 56 		this._setSelectionByRange(rng);
 57 	},
 58 
 59 	_setSelectionByRange: function(rng) {
 60 		var sel = this.sel();
 61 		sel.setBaseAndExtent(rng.startContainer, rng.startOffset, rng.endContainer, rng.endOffset);
 62 	}
 63 });
