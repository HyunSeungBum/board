1 /**
  2  * @requires XQuared.js
  3  * @requires Browser.js
  4  * @requires Editor.js
  5  * @requires plugin/Base.js
  6  */
  7 xq.plugin.EventLogPlugin = xq.Class(xq.plugin.Base,
  8 	/**
  9 	 * @name xq.plugin.EventLogPlugin
 10 	 * @lends xq.plugin.EventLogPlugin.prototype
 11 	 * @extends xq.plugin.Base
 12 	 * @constructor
 13 	 */
 14 	{
 15 	isEventListener: function() {return true;},
 16 	
 17 	onAfterLoad: function(xed) {
 18 		this.createLogWindow();
 19 	},
 20 	
 21 	onEditorStartInitialization: function(xed) {
 22 		this.log("Start initialization.");
 23 	},
 24 	onEditorInitialized: function(xed) {
 25 		this.log("Initialized.");
 26 	},
 27 	onEditorElementChanged: function(xed, from, to) {
 28 		this.log("Element changed from <" + (from ? from.nodeName : null) + "> to <" + (to ? to.nodeName : null) + ">.");
 29 	},
 30 	onEditorBeforeEvent: function(xed, e) {
 31 		this.log("Before event [" + e.type + "]");
 32 	},
 33 	onEditorAfterEvent: function(xed, e) {
 34 		this.log("After event [" + e.type + "]");
 35 	},
 36 	onEditorCurrentContentChanged: function(xed) {
 37 		this.log("Current content changed.");
 38 	},
 39 	onEditorStaticContentChanged: function(xed, content) {
 40 		this.log("Static content changed.");
 41 	},
 42 	onEditorCurrentEditModeChanged: function(xed, from, to) {
 43 		this.log("Edit mode changed from <" + from + "> to <" + to + ">.");
 44 	},
 45 	
 46 	
 47 	
 48 	createLogWindow: function() {
 49 		var wrapper = document.createElement("DIV");
 50 		wrapper.innerHTML = "<h2>Log</h2>";
 51 		wrapper.style.width = "500px";
 52 		document.body.appendChild(wrapper);
 53 		
 54 		this.logWindow = document.createElement("PRE");
 55 		this.logWindow.style.fontSize = "0.75em";
 56 		this.logWindow.style.height = "200px";
 57 		this.logWindow.style.overflow = "scroll";
 58 		this.logWindow.style.border = "1px solid black";
 59 		this.logWindow.style.padding = "2px";
 60 		wrapper.appendChild(this.logWindow);
 61 	},
 62 	
 63 	log: function(message) {
 64 		var line = document.createTextNode(this.getFormattedTime() + ": " + message);
 65 		this.logWindow.insertBefore(document.createElement("BR"), this.logWindow.firstChild);
 66 		this.logWindow.insertBefore(line, this.logWindow.firstChild);
 67 	},
 68 	
 69 	getFormattedTime: function() {
 70 		var date = new Date();
 71 		var time = date.toTimeString().split(" ")[0];
 72 		var msec = "000" + date.getMilliseconds();
 73 		msec = msec.substring(msec.length - 4);
 74 		
 75 		return time + "." + msec;
 76 	}
 77 });
