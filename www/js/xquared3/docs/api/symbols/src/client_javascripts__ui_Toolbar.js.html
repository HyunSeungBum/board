1 /**

  6 xq.ui.Toolbar = xq.Class(/** @lends xq.ui.Toolbar.prototype */{
  7 	/**
  8 	 * Editor's toolbar.
  9 	 *
 10      * @constructs
 11 	 */
 12 	initialize: function(xed, container, wrapper, buttonMap, buttonList, imagePath, structureAndStyleCollector) {
 13 	
 14 		xq.addToFinalizeQueue(this);
 15 		
 16 		this.xed = xed;
 17 		
 18 		if(typeof container === 'string') {
 19 			container = xq.$(container);
 20 		}
 21 		if(container && container.nodeType !== 1) {
 22 			throw "[container] is not an element";
 23 		}
 24 		
 25 		this.wrapper = wrapper;
 26 		this.doc = this.wrapper.ownerDocument;
 27 		this.buttonMap = buttonMap;
 28 		this.buttonList = buttonList;
 29 		this.imagePath = imagePath;
 30 		this.structureAndStyleCollector = structureAndStyleCollector;
 31 		
 32 		this.buttons = null;
 33 		this.anchorsCache = [];
 34 		this._scheduledUpdate = null;
 35 		
 36 		if(!container) {
 37 			this.create();
 38 			this._addStyleRules([
 39 				{selector:".xquared div.toolbar", rule:"background-image: url(" + imagePath + "toolbarBg.gif)"},
 40 				{selector:".xquared ul.buttons li", rule:"background-image: url(" + imagePath + "toolbarButtonBg.gif)"},
 41 				{selector:".xquared ul.buttons li.xq_separator", rule:"background-image: url(" + imagePath + "toolbarSeparator.gif)"}
 42 			]);
 43 		} else {
 44 			this.container = container;
 45 			
 46 			if(!this.dialogContainer)
 47 			{
 48 				// dialog container
 49 				var dialogs = this.doc.createElement('div');
 50 				dialogs.className = 'dialogs';
 51 				this.dialogContainer = dialogs;
 52 				this.wrapper.appendChild(dialogs);
 53 			}
 54 		}
 55 		xed.addListener({
 56 			onEditorInitialized: function() {
 57 				xq.observe(xed.getDoc(), 'mousedown', this._closeAllLightweight.bindAsEventListener(this));
 58 				xq.observe(document, 'mousedown', this._closeAllLightweight.bindAsEventListener(this));
 59 			}.bind(this)
 60 		});
 61 	},
 62 	
 63 	finalize: function() {
 64 		for(var i = 0; i < this.anchorsCache.length; i++) {
 65 			// TODO remove dependency to Editor
 66 			this.anchorsCache[i].xed = null;
 67 			this.anchorsCache[i].handler = null;
 68 			this.anchorsCache[i] = null;
 69 		}
 70 	
 71 		this.toolbarAnchorsCache = null;
 72 	},
 73 	
 74 	triggerUpdate: function() {
 75 		if(this._scheduledUpdate) return;
 76 		
 77 		this._scheduledUpdate = window.setTimeout(
 78 			function() {
 79 				this._scheduledUpdate = null;
 80 				var ss = this.structureAndStyleCollector();
 81 				if(ss) this.update(ss);
 82 			}.bind(this), 200
 83 		);
 84 	},
 85 	
 86 	/**
 87 	 * Updates all buttons' status. Override this to customize status L&F. Don't call this function directly. Use triggerUpdate() to call it indirectly.
 88 	 * 
 89 	 * @param {Object} structure and style information. see xq.rdom.Base.collectStructureAndStyle()
 90 	 */
 91 	update: function(info) {
 92 		if(!this.container) return;
 93 		if(!this.buttons) {
 94 			var classNames = [
 95 				"emphasis", "strongEmphasis", "underline", "strike", "superscription", "subscription",
 96 				"justifyLeft", "justifyCenter", "justifyRight", "justifyBoth",
 97 				"unorderedList", "orderedList", "code",
 98 				"paragraph", "heading1", "heading2", "heading3", "heading4", "heading5", "heading6"
 99 			];
100 			
101 			this.buttons = {};
102 			
103 			for(var i = 0; i < classNames.length; i++) {
104 				var found = xq.getElementsByClassName(this.container, classNames[i]);
105 				var button = found && found.length > 0 ? found[0] : null;
106 				if(button) this.buttons[classNames[i]] = button;
107 			}
108 		}
109 		
110 		var buttons = this.buttons;
111 		this._updateButtonStatus('emphasis', info.em);
112 		this._updateButtonStatus('strongEmphasis', info.strong);
113 		this._updateButtonStatus('underline', info.underline);
114 		this._updateButtonStatus('strike', info.strike);
115 		this._updateButtonStatus('superscription', info.superscription);
116 		this._updateButtonStatus('subscription', info.subscription);
117 		
118 		this._updateButtonStatus('justifyLeft', info.justification === 'left');
119 		this._updateButtonStatus('justifyCenter', info.justification === 'center');
120 		this._updateButtonStatus('justifyRight', info.justification === 'right');
121 		this._updateButtonStatus('justifyBoth', info.justification === 'justify');
122 		
123 		this._updateButtonStatus('orderedList', info.list === 'OL');
124 		this._updateButtonStatus('unorderedList', info.list === 'UL');
125 		this._updateButtonStatus('code', info.list === 'CODE');
126 		
127 		this._updateButtonStatus('paragraph', info.block === 'P');
128 		this._updateButtonStatus('heading1', info.block === 'H1');
129 		this._updateButtonStatus('heading2', info.block === 'H2');
130 		this._updateButtonStatus('heading3', info.block === 'H3');
131 		this._updateButtonStatus('heading4', info.block === 'H4');
132 		this._updateButtonStatus('heading5', info.block === 'H5');
133 		this._updateButtonStatus('heading6', info.block === 'H6');
134 	},
135 
136 	/**
137 	 * Enables all buttons
138 	 *
139 	 * @param {Array} [exceptions] array of string containing classnames to exclude
140 	 */
141 	enableButtons: function(exceptions) {
142 		if(!this.container) return;
143 		
144 		this._execForAllButtons(exceptions, function(li, exception) {
145 			li.firstChild.className = !exception ? '' : 'disabled';
146 		});
147 
148 		// @WORKAROUND: Image icon disappears without following code:
149 		if(xq.Browser.isIE6) {
150 			this.container.style.display = 'none';
151 			setTimeout(function() {this.container.style.display = 'block';}.bind(this), 0);
152 		}
153 	},
154 	
155 	/**
156 	 * Disables all buttons
157 	 *
158 	 * @param {Array} [exceptions] array of string containing classnames to exclude
159 	 */
160 	disableButtons: function(exceptions) { 
161 		this._execForAllButtons(exceptions, function(li, exception) {
162 			li.firstChild.className = exception ? '' : 'disabled';
163 		});
164 	},

166 	/**
167 	 * Creates toolbar element
168 	 */
169 	create: function() {
170 		// outmost container
171 		this.container = this.doc.createElement('div');
172 		this.container.className = 'toolbar';
173 		
174 		// button container
175 		var buttons = this.doc.createElement('ul');
176 		buttons.className = 'buttons';
177 		this.container.appendChild(buttons);
178 		
179 		// dialog container
180 		var dialogs = this.doc.createElement('div');
181 		dialogs.className = 'dialogs';
182 		this.dialogContainer = dialogs;
183 		this.wrapper.appendChild(dialogs);
184 		
185 		if(this.buttonList.length !== 0)
186 		{
187 			var btnListLen = this.buttonList.length;
188 			for(var i = 0; i < btnListLen; i++)
189 			{
190 				if(this.buttonList[i] == "separator" )
191 				{
192 					continue;
193 				}
194 				
195 				var buttonConfig = this.buttonList[i];
196 				var li = this.doc.createElement('li');
197 				buttons.appendChild(li);
198 				li.className = buttonConfig.className;
199 				
200 				if(typeof this.buttonList[i-1] !== "undefined" && this.buttonList[i-1] == "separator" )
201 				{
202 					li.className += ' xq_separator';
203 				}
204 				
205 				var span = this.doc.createElement('span');
206 				li.appendChild(span);
207 				
208 				if(buttonConfig.list) {
209 					this._createDropdown(buttonConfig, span);
210 				} else {
211 					this._createButton(buttonConfig, span);
212 				}
213 			}
214 		}
215 		else if(this.buttonMap)
216 		{
217 			// Generate buttons from map and append it to button container
218 			for(var i = 0; i < this.buttonMap.length; i++) {
219 				for(var j = 0; j < this.buttonMap[i].length; j++) {
220 					var buttonConfig = this.buttonMap[i][j];
221 					var li = this.doc.createElement('li');
222 					buttons.appendChild(li);
223 					li.className = buttonConfig.className;
224 					
225 					var span = this.doc.createElement('span');
226 					li.appendChild(span);
227 					
228 					if(buttonConfig.list) {
229 						this._createDropdown(buttonConfig, span);
230 					} else {
231 						this._createButton(buttonConfig, span);
232 					}
233 					
234 					if(j === 0 && i !== 0) li.className += ' xq_separator';
235 				}
236 			}
237 		}
238 		
239 		this.wrapper.appendChild(this.container);
240 	},
241 
242 	_createButton: function(buttonConfig, span) {
243 		var a = this.doc.createElement('a');
244 		span.appendChild(a);
245 		
246 		a.href = "#";
247 		a.title = buttonConfig.title;
248 		if (buttonConfig.handler){
249 			a.handler = buttonConfig.handler;
250 			xq.observe(a, 'click', this._clickHandler.bindAsEventListener(this));
251 		}
252 		
253 		this.anchorsCache.push(a);
254 		
255 		xq.observe(a, 'mousedown', xq.cancelHandler);
256 
257 		var img = this.doc.createElement('img');
258 		a.appendChild(img);
259 		img.className = buttonConfig.className;
260 		img.src = this.imagePath + buttonConfig.className + '.gif';
261 		
262 		if(buttonConfig.title)
263 		{
264 			img.alt = buttonConfig.title;
265 		}
266 		else
267 		{
268 			img.alt = buttonConfig.className;
269 		}
270 
271 		return a;
272 	},
273 	
274 	_createDropdown: function(buttonConfig, span) {
275 		// Create button
276 		var btn = this._createButton(buttonConfig, span);
277 		btn.items = buttonConfig.list;
278 		
279 		xq.observe(btn, 'click', this._openDropdownDialog.bindAsEventListener(this));
280 		
281 		// Create dialog
282 		var dialog = this.doc.createElement('DIV');
283 		dialog.id = buttonConfig.className + "Dialog";
284 		dialog.className = "xqFormDialog lightweight";
285 		dialog.style.display = 'none';
286 		
287 		var title = this.doc.createElement('H3');
288 		title.innerHTML = buttonConfig.title;
289 		dialog.appendChild(title);
290 
291 		var dialogContent = this.doc.createElement('DIV');
292 		dialogContent.className = 'dialog-content';
293 		
294 		var ul = this.doc.createElement('UL');
295 		ul.className = "item-list";
296 		
297 		for (var i = 0; i < btn.items.length; i++) {
298 			var item = btn.items[i];
299 			var li = this.doc.createElement('LI');
300 			var anchor = this.doc.createElement('A');
301 			li.appendChild(anchor);
302 			if (item.html) {
303 				if (buttonConfig.className == 'emoticon') {
304 					var emoticon = this.doc.createElement('IMG');
305 					emoticon.src = this.xed.config.imagePathForEmoticon + item.html;
306 					emoticon.alt = item.html;
307 					anchor.appendChild(emoticon);				
308 				} else {
309 					anchor.innerHTML = decodeURIComponent(item.html);
310 				}
311 			}
312 			
313 			anchor.href = "#";
314 			anchor.handler = item.handler;
315 			
316 			for (attr in item.style){
317 				anchor.style[attr] = item.style[attr];
318 			}
319 			xq.observe(anchor, 'click', xq.cancelHandler);
320 			xq.observe(anchor, 'mouseup', this._closeAllLightweight.bindAsEventListener(this));
321 			xq.observe(anchor, 'mousedown', this._clickHandler.bindAsEventListener(this));
322 			
323 			ul.appendChild(li);
324 		}
325 		
326 		dialogContent.appendChild(ul);
327 		dialog.appendChild(dialogContent);
328 		this.dialogContainer.appendChild(dialog);
329 		span.appendChild(btn);
330 	},
331 	
332 	_openDropdownDialog: function(e){
333 		this._closeAllLightweight(e);
334 		
335 		var src = e.target || e.srcElement;
336 		this.xed.lastAnchor = src;
337 		var dialog = xq.$(src.className + "Dialog");
338 		
339 		if (dialog) {
340 			dialog.style.display = 'block';
341 			dialog.style.top = this.container.offsetTop + this.container.offsetHeight + 'px';
342 			dialog.style.left = this.container.offsetLeft + src.parentNode.offsetLeft + 'px';
343 		} 
344 		xq.stopEvent(e);
345 		return false;
346 	},
347 	
348 	_closeAllLightweight: function(e){
349 		if(e)
350 		{
351 			var src = e.target || e.srcElement;
352 			
353 			var linkDlg = xq.$('linkDialog');
354 			
355 			if( (linkDlg !== null && linkDlg.style.display !== 'none') || src.id.indexOf("extForeColor") !== -1 || src.className.indexOf("jscolor") !== -1)
356 			{
357 				return false;
358 			}
359 		}
360 		
361 		var dialogs = xq.getElementsByClassName(this.dialogContainer, 'lightweight');
362 		for (var i = 0; i < dialogs.length; i++){
363 			dialogs[i].style.display = "none";
364 		}
365 	},
366 	_clickHandler: function(e) {
367 		var src = e.target || e.srcElement;
368 		
369 		while(src.nodeName !== "A") src = src.parentNode;
370 		
371 		if(xq.hasClassName(src.parentNode, 'disabled') || xq.hasClassName(this.container, 'disabled')) {
372 			xq.stopEvent(e);
373 			return false;
374 		}
375 		
376 		var handler = src.handler;
377 		var xed = this.xed;
378 		xed.focus();
379 		if(typeof handler === "function") {
380 			handler(this);
381 		} else {
382 			eval(handler);
383 		}
384 		
385 		xq.stopEvent(e);
386 		return false;
387 	},
388 
389 	_updateButtonStatus: function(className, selected) {
390 		var button = this.buttons[className];
391 		if(button) {
392 			var newClassName = selected ? 'selected' : '';
393 			var target = button.firstChild.firstChild;
394 			if(target.className !== newClassName) target.className = newClassName;
395 		}
396 	},
397 	
398 	_execForAllButtons: function(exceptions, exec) {
399 		if(!this.container) return;
400 		exceptions = exceptions || [];
401 		
402 		var lis = this.container.getElementsByTagName('LI');
403 		for(var i = 0; i < lis.length; i++) {
404 			var className = lis[i].className.split(" ").find(function(name) {return name !== 'xq_separator'});
405 			var exception = exceptions.indexOf(className) !== -1;
406 			exec(lis[i], exception);
407 		}
408 	},
409 	
410 	_addStyleRules: function(rules) {
411 		
412 		if(!this.dynamicStyle) {
413 			if(xq.Browser.isTrident) {
414 			    this.dynamicStyle = this.doc.createStyleSheet();
415 			} else {
416 	    		var style = this.doc.createElement('style');
417 	    		this.doc.body.appendChild(style);
418 		    	this.dynamicStyle = xq.$A(this.doc.styleSheets).last();
419 			}
420 		}
421 		
422 		for(var i = 0; i < rules.length; i++) {
423 			var rule = rules[i];
424 					
425 			if(xq.Browser.isTrident) {
426 				
427 				this.dynamicStyle.addRule(rules[i].selector, rules[i].rule);
428 			} else {
429 		    	this.dynamicStyle.insertRule(rules[i].selector + " {" + rules[i].rule + "}", this.dynamicStyle.cssRules.length);
430 	    	}
431 		}
432 	}
