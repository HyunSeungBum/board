1 /**
  2  * @requires XQuared.js
  3  * @requires rdom/Factory.js
  4  */
  5 xq.EditHistory = xq.Class(/** @lends xq.EditHistory.prototype */{
  6     /**
  7 	 * Manages editing history and performs UNDO/REDO.
  8 	 *
  9      * @constructs
 10 	 * @param {xq.rdom.Base} rdom Base instance
 11 	 * @param {Number} [max=100] maximum UNDO buffer size.
 12 	 */
 13 	initialize: function(rdom, max) {
 14 		xq.addToFinalizeQueue(this);
 15 		if (!rdom) throw "IllegalArgumentException";
 16 
 17 		this.disabled = false;
 18 		this.max = max || 100;
 19 		this.rdom = rdom;
 20 		this.index = -1;
 21 		this.queue = [];
 22 		
 23 		this.lastModified = Date.get();
 24 	},
 25 	getLastModifiedDate: function() {
 26 		return this.lastModified;
 27 	},
 28 	isUndoable: function() {
 29 		return this.queue.length > 0 && this.index > 0;
 30 	},
 31 	isRedoable: function() {
 32 		return this.queue.length > 0 && this.index < this.queue.length - 1;
 33 	},
 34 	disable: function() {
 35 		this.disabled = true;
 36 	},
 37 	enable: function() {
 38 		this.disabled = false;
 39 	},
 40 	undo: function() {
 41 		this.pushContent();
 42 		
 43 		if (this.isUndoable()) {
 44 			this.index--;
 45 			this.popContent();
 46 			return true;
 47 		} else {
 48 			return false;
 49 		}
 50 	},
 51 	redo: function() {
 52 		if (this.isRedoable()) {
 53 			this.index++;
 54 			this.popContent();
 55 			return true;
 56 		} else {
 57 			return false;
 58 		}
 59 	},
 60 	onCommand: function() {
 61 		this.lastModified = Date.get();
 62 		if(this.disabled) return false;
 63 		
 64 		return this.pushContent();
 65 	},
 66 	onEvent: function(event) {
 67 		this.lastModified = Date.get();
 68 		if(this.disabled) return false;
 69 		
 70 		var arrowKeys = [33,34,35,36,37,39];
 71 		// @WORKAROUND: Mac에서 화살표 up/down 누를 때 pushContent 하면 캐럿이 튄다
 72 		if(!xq.Browser.isMac) arrowKeys.push(38,40);
 73 		
 74 		// ignore some event types
 75 		if(['blur', 'mouseup'].indexOf(event.type) !== -1) return false;
 76 		
 77 		// ignore normal keys
 78 		if('keydown' === event.type && !(event.ctrlKey || event.metaKey)) return false;
 79 		if(['keydown', 'keyup', 'keypress'].indexOf(event.type) !== -1 && !event.ctrlKey && !event.altKey && !event.metaKey && arrowKeys.indexOf(event.keyCode) === -1) return false;
 80 		if(['keydown', 'keyup', 'keypress'].indexOf(event.type) !== -1 && (event.ctrlKey || event.metaKey) && [89,90].indexOf(event.keyCode) !== -1) return false;
 81 		
 82 		// ignore ctrl/shift/alt/meta keys
 83 		if([16,17,18,224].indexOf(event.keyCode) !== -1) return false;
 84 		
 85 		return this.pushContent();
 86 	},
 87 	popContent: function() {
 88 		this.lastModified = Date.get();
 89 		var entry = this.queue[this.index];
 90 		if (entry.caret > 0) {
 91 			var html=entry.html.substring(0, entry.caret) + '<span id="caret_marker_eh"></span>' + entry.html.substring(entry.caret);
 92 			this.rdom.getRoot().innerHTML = html;
 93 		} else {
 94 			this.rdom.getRoot().innerHTML = entry.html;
 95 		}
 96 		this.restoreCaret();
 97 	},
 98 	pushContent: function(ignoreCaret) {
 99 		if(xq.Browser.isTrident && !ignoreCaret && !this.rdom.hasFocus()) return false;
100 		if(!this.rdom.getCurrentElement()) return false;
101 
102 		var html = this.rdom.getRoot().innerHTML;
103 		if(html === (this.queue[this.index] ? this.queue[this.index].html : null)) return false;
104 		
105 		var caret = ignoreCaret ? -1 : this.saveCaret();
106 		
107 		if(this.queue.length >= this.max) {
108 			this.queue.shift();
109 		} else {
110 			this.index++;
111 		}
112 		
113 		this.queue.splice(this.index, this.queue.length - this.index, {html:html, caret:caret});
114 		return true;
115 	},
116 	clear: function() {
117 		this.index = -1;
118 		this.queue = [];
119 		this.pushContent(true);
120 	},
121 	saveCaret: function() {
122 		if(this.rdom.hasSelection()) return null;
123 		
124 		var marker = this.rdom.pushMarker();
125 		
126 		var str = xq.Browser.isTrident ? '<SPAN class='+marker.className : '<span class="'+marker.className+'"';
127 		var caret = this.rdom.getRoot().innerHTML.indexOf(str);
128 		
129 		this.rdom.popMarker(true);
130 		
131 		return caret;
132 	},
133 	restoreCaret: function() {
134 		var marker = this.rdom.$('caret_marker_eh');
135 		
136 		if(marker) {
137 			this.rdom.selectElement(marker, true);
138 			this.rdom.collapseSelection(false);
139 			this.rdom.deleteNode(marker);
140 		} else {
141 			var node = this.rdom.tree.findForward(this.rdom.getRoot(), function(node) {
142 				return this.isBlock(node) && !this.hasBlocks(node);
143 			}.bind(this.rdom.tree));
144 			this.rdom.selectElement(node, false);
145 			this.rdom.collapseSelection(false);
146 			
147 		}
148 	}
149 });
