 27 	/**
 28 	 * Find common ancestor(parent) and his immediate children(left and right).<br />
 29 	 *<br />
 30 	 * A --- B -+- C -+- D -+- E<br />
 31 	 *          |<br />
 32 	 *          +- F -+- G<br />
 33 	 *<br />
 34 	 * For example:<br />
 35 	 * > findCommonAncestorAndImmediateChildrenOf("E", "G")<br />
 36 	 *<br />
 37 	 * will return<br />
 38 	 *<br />
 39 	 * > {parent:"B", left:"C", right:"F"}
 40 	 */
 41 	findCommonAncestorAndImmediateChildrenOf: function(left, right) {
 42 		if(left.parentNode === right.parentNode) {
 43 			return {
 44 				left:left,
 45 				right:right,
 46 				parent:left.parentNode
 47 			};
 48 		} else {
 49 			var parentsOfLeft = this.collectParentsOf(left, true);
 50 			var parentsOfRight = this.collectParentsOf(right, true);
 51 			var ca = this.getCommonAncestor(parentsOfLeft, parentsOfRight);
 52 	
 53 			var leftAncestor = parentsOfLeft.find(function(node) {return node.parentNode === ca});
 54 			var rightAncestor = parentsOfRight.find(function(node) {return node.parentNode === ca});
 55 			
 56 			return {
 57 				left:leftAncestor,
 58 				right:rightAncestor,
 59 				parent:ca
 60 			};
 61 		}
 62 	},
 63 	
 64 	/**
 65 	 * Find leaves at edge.<br />
 66 	 *<br />
 67 	 * A --- B -+- C -+- D -+- E<br />
 68 	 *          |<br />
 69 	 *          +- F -+- G<br />
 70 	 *<br />
 71 	 * For example:<br />
 72 	 * > getLeavesAtEdge("A")<br />
 73 	 *<br />
 74 	 * will return<br />
 75 	 *<br />
 76 	 * > ["E", "G"]
 77 	 */
 78 	getLeavesAtEdge: function(element) {
 79 		if(!element.hasChildNodes()) return [null, null];
 80 		
 81 		var findLeft = function(el) {
 82 			for (var i = 0; i < el.childNodes.length; i++) {
 83 				if (el.childNodes[i].nodeType === 1 && this.isBlock(el.childNodes[i])) return findLeft(el.childNodes[i]);
 84 			}
 85 			return el;
 86 		}.bind(this);
 87 		
 88 		var findRight=function(el) {
 89 			for (var i = el.childNodes.length; i--;) {
 90 				if (el.childNodes[i].nodeType === 1 && this.isBlock(el.childNodes[i])) return findRight(el.childNodes[i]);
 91 			}
 92 			return el;
 93 		}.bind(this);
 94 		
 95 		var left = findLeft(element);
 96 		var right = findRight(element);
 97 		
 98 		return [left === element ? null : left, right === element ? null : right];
 99 	},

101 	getCommonAncestor: function(parents1, parents2) {
102 		for(var i = 0; i < parents1.length; i++) {
103 			for(var j = 0; j < parents2.length; j++) {
104 				if(parents1[i] === parents2[j]) return parents1[i];
105 			}
106 		}
107 	},

109 	collectParentsOf: function(node, includeSelf, exitCondition) {
110 		var parents = [];
111 		if(includeSelf) parents.push(node);
112 		
113 		while((node = node.parentNode) && (node.nodeName !== "HTML") && !(typeof exitCondition === "function" && exitCondition(node))) parents.push(node);
114 		return parents;
115 	},
116 	
117 	isDescendantOf: function(parent, child) {
118 		if(parent.length > 0) {
119 			for(var i = 0; i < parent.length; i++) {
120 				if(this.isDescendantOf(parent[i], child)) return true;
121 			}
122 			return false;
123 		}
124 		
125 		if(parent === child) return false;
126 		
127 	    while (child = child.parentNode)
128 	      if (child === parent) return true;
129 	    return false;
130 	},

132 	/**
133 	 * Perform tree walking (foreward)
134 	 */
135 	walkForward: function(node) {
136 		var target = node.firstChild;
137 		if(target) return target;
138 		
139 		// intentional assignment for micro performance turing
140 		if(target = node.nextSibling) return target;
141 		
142 		while(node = node.parentNode) {
143 			// intentional assignment for micro performance turing
144 			if(target = node.nextSibling) return target;
145 		}
146 		
147 		return null;
148 	},
149 	
150 	/**
151 	 * Perform tree walking (backward)
152 	 */
153 	walkBackward: function(node) {
154 		if(node.previousSibling) {
155 			node = node.previousSibling;
156 			while(node.hasChildNodes()) {node = node.lastChild;}
157 			return node;
158 		}
159 		
160 		return node.parentNode;
161 	},
162 	
163 	/**
164 	 * Perform tree walking (to next siblings)
165 	 */
166 	walkNext: function(node) {return node.nextSibling},
167 	
168 	/**
169 	 * Perform tree walking (to next siblings)
170 	 */
171 	walkPrev: function(node) {return node.previousSibling},

195 	_check: function(start, direction, target) {
196 		if(start === target) return false;
197 		
198 		while(start = direction(start)) {
199 			if(start === target) return true;
200 		}
201 		return false;
202 	},
203 	
204 	_find: function(start, direction, condition, exitCondition) {
205 		while(start = direction(start)) {
206 			if(exitCondition && exitCondition(start)) return null;
207 			if(condition(start)) return start;
208 		}
209 		return null;
210 	},

212 	/**
213 	 * Walks Forward through DOM tree from start to end, and collects all nodes that matches with a filter.
214 	 * If no filter provided, it just collects all nodes.
215 	 *
216 	 * @param {Element} start Starting element.
217 	 * @param {Element} end Ending element.
218 	 * @param {Function} filter A filter function.
219 	 */
220 	collectNodesBetween: function(start, end, filter) {
221 		if(start === end) return [start, end].findAll(filter || function() {return true});
222 		
223 		var nodes = this.collectForward(start, function(node) {return node === end}, filter);
224 		if(
225 			start !== end &&
226 			typeof filter === "function" &&
227 			filter(end)
228 		) nodes.push(end);
229 		
230 		return nodes;
231 	},
232

234 		return this.collect(start, this.walkForward, exitCondition, filter);
235 	},

238 		return this.collect(start, this.walkBackward, exitCondition, filter);
239 	},

242 		return this.collect(start, this.walkNext, exitCondition, filter);
243 	},

246 		return this.collect(start, this.walkPrev, exitCondition, filter);
247 	},

249 	collect: function(start, next, exitCondition, filter) {
250 		var nodes = [start];
251 
252 		while(true) {
253 			start = next(start);
254 			if(
255 				(start === null) ||
256 				(typeof exitCondition === "function" && exitCondition(start))
257 			) break;
258 			
259 			nodes.push(start);
260 		}
261 
262 		return (typeof filter === "function") ? nodes.findAll(filter) : nodes;
263 	},

265 	hasBlocks: function(element) {
266 		var nodes = element.childNodes;
267 		for(var i = 0; i < nodes.length; i++) {
268 			if(this.isBlock(nodes[i])) return true;
269 		}
270 		return false;
271 	},

273 	hasMixedContents: function(element) {
274 		if(!this.isBlock(element)) return false;
275 		if(!this.isBlockContainer(element)) return false;
276 		
277 		var hasTextOrInline = false;
278 		var hasBlock = false;
279 		for(var i = 0; i < element.childNodes.length; i++) {
280 			var node = element.childNodes[i];
281 			if(!hasTextOrInline && this.isTextOrInlineNode(node)) hasTextOrInline = true;
282 			if(!hasBlock && this.isBlock(node)) hasBlock = true;
283 			
284 			if(hasTextOrInline && hasBlock) break;
285 		}
286 		if(!hasTextOrInline || !hasBlock) return false;
287 		
288 		return true;
289 	},
