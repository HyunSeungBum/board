 27 	/**
 28 	 * Perform validation on given element
 29 	 *
 30 	 * @param {Element} element Target element. It is not affected by validation.
 31 	 *
 32 	 * @returns {String} Validated HTML string
 33 	 */
 34 	validate: function(element, dontClone) {
 35 		// DOM validation
 36 		element = dontClone ? element : element.cloneNode(true);
 37 		this._fireOnBeforeDomValidation(element);
 38 		this.validateDom(element);
 39 		this._fireOnAfterDomValidation(element);
 40 		
 41 		// String validation
 42 		var html = {value: element.innerHTML};
 43 		this._fireOnBeforeStringValidation(html);
 44 		html.value = this.validateString(html.value);
 45 		this._fireOnAfterStringValidation(html);
 46 		
 47 		return html.value;
 48 	},

 53 	/**
 54 	 * Perform invalidation on given element to make the designmode works well.
 55 	 *
 56 	 * @param {String} html HTML string.
 57 	 * @returns {String} Invalidated HTML string
 58 	 */
 59 	invalidate: function(html) {
 60 		// Preprocessing
 61 		var html = {value: html};
 62 		this._fireOnPreprocessing(html);
 63 		
 64 		// DOM invalidation
 65 		var element = document.createElement("DIV");
 66 		element.innerHTML = html.value;
 67 		this._fireOnBeforeDomInvalidation(element);
 68 		this.invalidateDom(element);
 69 		this._fireOnAfterDomInvalidation(element);
 70 		
 71 		// String invalidation
 72 		html.value = element.innerHTML;
 73 		this._fireOnBeforeStringInvalidation(html);
 74 		html.value = this.invalidateString(html.value);
 75 		this._fireOnAfterStringInvalidation(html);
 76 		
 77 		return html.value;
 78 	},

 87 	invalidateStrikesAndUnderlines: function(element) {
 88 		var rdom = xq.rdom.Base.createInstance();
 89 		rdom.setRoot(element);
 90 
 91 		var nameOfClassName = xq.Browser.isTrident ? "className" : "class";
 92 		
 93 		var underlines = xq.getElementsByClassName(rdom.getRoot(), "underline", "em");
 94 		var pUnderline = xq.compilePattern("(^|\\s)underline($|\\s)");
 95 		var lenOfUnderlines = underlines.length;
 96 		for(var i = 0; i < lenOfUnderlines; i++) {
 97 			rdom.replaceTag("u", underlines[i]).removeAttribute(nameOfClassName);
 98 		}
 99 		
100 		var strikes = xq.getElementsByClassName(rdom.getRoot(), "strike", "span")
101 		var pStrike = xq.compilePattern("(^|\\s)strike($|\\s)");
102 		var lenOfStrikes = strikes.length;
103 		for(var i = 0; i < lenOfStrikes; i++) {
104 			rdom.replaceTag("strike", strikes[i]).removeAttribute(nameOfClassName);
105 		}

130 	validateFont: function(element) {
131 		var rdom = xq.rdom.Base.createInstance();
132 		rdom.setRoot(element);
133 		
134 		// It should be reversed to deal with nested elements
135 		var fonts = element.getElementsByTagName('FONT');
136 		var fontSizes = ["x-small", "small", "medium", "large", "x-large", "xx-large"];
137 		var len = fonts.length - 1;
138 		for(var i = len; i >= 0; i--) {
139 			var font = fonts[i];
140 			var color = font.getAttribute('color');
141 			var backgroundColor = font.style.backgroundColor;
142 			var face = font.getAttribute('face');
143 			var size = fontSizes[parseInt(font.getAttribute('size')) % 8 - 1];
144 			
145 			if(color || backgroundColor || face || size) {
146 				var span = rdom.replaceTag("span", font);
147 				span.removeAttribute('color');
148 				span.removeAttribute('face');
149 				span.removeAttribute('size');
150 				
151 				if(color) span.style.color = color;
152 				if(backgroundColor) span.style.backgroundColor = backgroundColor;
153 				if(face) span.style.fontFamily = face;
154 				if(size) span.style.fontSize = size;
155 			}
156 		}
157 	},

159 	invalidateFont: function(element) {
160 		var rdom = xq.rdom.Base.createInstance();
161 		rdom.setRoot(element);
162 
163 		// It should be reversed to deal with nested elements
164 		var spans = element.getElementsByTagName('SPAN');
165 		var fontSizes = {"x-small":1, "small":2, "medium":3, "large":4, "x-large":5, "xx-large":6};
166 		var len = spans.length - 1;
167 		for(var i = len; i >= 0; i--) {
168 			var span = spans[i];
169 			if(span.className === "strike") continue;
170 			
171 			var color = span.style.color;
172 			var backgroundColor = span.style.backgroundColor;
173 			var face = span.style.fontFamily;
174 			var size = fontSizes[span.style.fontSize];
175 			
176 			if(color || backgroundColor || face || size) {
177 				var font = rdom.replaceTag("font", span);
178 				font.style.cssText = "";
179 				
180 				if(color) font.setAttribute('color', this.asRGB(color));
181 				if(backgroundColor) font.style.backgroundColor = backgroundColor;
182 				if(face) font.setAttribute('face', face);
183 				if(size) font.setAttribute('size', size);
184 			}
185 		}

188 	asRGB: function(color) {
189 		if(color.indexOf("#") === 0) return color;
190 		
191 		var m = this.pRGB.exec(color);
192 		if(!m) return color;
193 		
194 		var r = Number(m[1]).toString(16);
195 		var g = Number(m[2]).toString(16);
196 		var b = Number(m[3]).toString(16);
197 		
198 		if(r.length === 1) r = "0" + r;
199 		if(g.length === 1) g = "0" + g;
200 		if(b.length === 1) b = "0" + b;
201 		
202 		return "#" + r + g + b;
203 	},

225 	applyWhitelist: function(content) {
226 		var whitelist = this.whitelist;
227 		var allowedAttrs = null;
228 		
229 		var p1 = xq.compilePattern("(^|\\s\")([^\"=]+)(\\s|$)", "g");
230 		var p2 = xq.compilePattern("(\\S+?)=\"[^\"]*\"", "g");
231 		return content.replace(new RegExp("(</?)([^>]+?)(>|\\s+([^>]*?)(\\s?/?)>)", "g"), function(str, head, tag, tail, attrs, selfClosing) {
232 			allowedAttrs = whitelist[tag];
233 			if(!allowedAttrs) return '';
234 			
235 			if(attrs) {
236 				if(xq.Browser.isTrident) attrs = attrs.replace(p1, '$1$2="$2"$3');
237 				
238 				var sb = [];
239 				var m = attrs.match(p2);
240 				for(var i = 0; i < m.length; i++) {
241 					var name = m[i].split('=')[0];
242 					if(allowedAttrs.indexOf(name.toLowerCase()) !== -1) sb.push(m[i]);
243 				}
244 				
245 				if(sb.length) {
246 					attrs = sb.join(' ');
247 					return head + tag + ' ' + attrs + selfClosing + '>';
248 				} else {
249 					return head + tag + selfClosing + '>';
250 				}
251 			} else {
252 				return str;
253 			}
254 		});
255 	},

258 	makeUrlsRelative: function(content) {
259 		var curUrl = this.curUrl;
260 		var urlParts = this.curUrlParts;
261 		
262 		var p1 = xq.compilePattern("(href|src)=\"([^\"]+)\"", "g");
263 		var p2 = xq.compilePattern("^\\w+://");
264 		
265 		// 1. find attributes and...
266 		return content.replace(/(<\w+\s+)(\/|([^>]+?)(\/?))>/g, function(str, head, ignored, attrs, tail) {
267 			if(attrs) {
268 				// 2. validate URL part
269 				attrs = attrs.replace(p1, function(str, name, url) {
270 					// 3. first, make it absolute
271 					var abs = null;
272 					if(url.charAt(0) === '#') {
273 						abs = urlParts.includeQuery + url;
274 					} else if(url.charAt(0) === '?') {
275 						abs = urlParts.includePath + url;
276 					} else if(url.charAt(0) === '/') {
277 						abs = urlParts.includeHost + url;
278 					} else if(url.match(p2)) {
279 						abs = url;
280 					} else {
281 						abs = urlParts.includeBase + url;
282 					}
283 					
284 					// 4. make it relative by removing same part
285 					var rel = abs;
286 					
287 					if(abs === urlParts.includeHost) {
288 						rel = "/";
289 					} else if(abs.indexOf(urlParts.includeQuery) === 0) {
290 						rel = abs.substring(urlParts.includeQuery.length);
291 					} else if(abs.indexOf(urlParts.includePath) === 0) {
292 						rel = abs.substring(urlParts.includePath.length);
293 					} else if(abs.indexOf(urlParts.includeBase) === 0) {
294 						rel = abs.substring(urlParts.includeBase.length);
295 					} else if(abs.indexOf(urlParts.includeHost) === 0) {
296 						rel = abs.substring(urlParts.includeHost.length);
297 					}
298 					
299 					if(rel === '') rel = '#';
300 					
301 					return name + '="' + rel + '"';
302 				});
303 				
304 				return head + attrs + tail + '>';
305 			} else {
306 				return str;
307 			}
308 		});
309 		
310 		return content;
311 	},
