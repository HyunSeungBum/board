1 /**
  2  * Creates and returns instance of browser specific implementation.
  3  * 
  4  * @requires XQuared.js
  5  * @requires rdom/Base.js
  6  * @requires rdom/Trident.js
  7  * @requires rdom/Gecko.js
  8  * @requires rdom/Webkit.js
  9  */
 10 xq.rdom.Base.createInstance = function() {
 11 	if(xq.Browser.isTrident) {
 12 		return new xq.rdom.Trident();
 13 	} else if(xq.Browser.isWebkit) {
 14 		return new xq.rdom.Webkit();
 15 	} else {
 16 		return new xq.rdom.Gecko();
 17 	}
 18 }
