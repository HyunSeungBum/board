1 /**
  2  * @requires XQuared.js
  3  * @requires Browser.js
  4  * @requires Editor.js
  5  * 
  6  * @requires macro/Factory.js
  7  * @requires Layer.js
  8  * @requires Json2.js
  9  * 
 10  * @requires plugin/Base.js
 11  */
 12 xq.plugin.MacroPlugin = xq.Class(xq.plugin.Base,
 13 	/**
 14 	 * @name xq.plugin.MacroPlugin
 15 	 * @lends xq.plugin.MacroPlugin.prototype
 16 	 * @extends xq.plugin.Base
 17 	 * @constructor
 18 	 */
 19 	{
 20 	isEventListener: function() {return true;},
 21 	
 22 	onAfterLoad: function(xed) {
 23 		this.xed = xed;
 24 		this.xed.config.macroIds = [];
 25 		this.layers = [];
 26 	},
 27 	
 28 	onEditorStartInitialization: function(xed) {
 29 		this.xed.validator.addListener(this);
 30 		
 31 		this.xed.macroFactory = new xq.macro.Factory(this.xed.config.imagePathForContent + 'placeholder.gif');
 32 		for(var i = 0; i < this.xed.config.macroIds.length; i++) {
 33 			this.xed.macroFactory.register(this.xed.config.macroIds[i]);
 34 		}
 35 
 36 		xed.timer.register(this.updateLayers.bind(this), 100);
 37 		xed.timer.register(this.updateLayerList.bind(this), 2000);
 38 	},
 39 	
 40 	/**
 41 	 * @param {Element} [placeHolder] place holder element
 42 	 */
 43 	attachMacro: function(element) {
 44 		var longdesc = element.getAttribute("longdesc") || element.longdesc;
 45 		var def = JSON.parse(unescape(longdesc));
 46 		var macro = this.xed.macroFactory.createMacroFromDefinition(def);
 47 		var layer = new xq.Layer(this.xed, element, macro.createHtml());
 48 		macro.onLayerInitialzied(layer);
 49 		this.layers.push(layer);
 50 	},
 51 	
 52 	isAttachedPlaceHolder: function(element) {
 53 		for(var i = 0; i < this.layers.length; i++) {
 54 			if(this.layers[i].element === element) return true;
 55 		}
 56 		return false;
 57 	},
 58 
 59 	updateLayerList: function() {
 60 		if(this.xed.getCurrentEditMode() !== 'wysiwyg') {
 61 			for(var i = 0; i < this.layers.length; i++) {
 62 				this.layers[i].detach();
 63 			}
 64 			this.layers = [];
 65 		} else {
 66 			var placeHolders = xq.getElementsByClassName(this.xed.rdom.getRoot(), "xqlayer", xq.Browser.isTrident ? "img" : null);
 67 			for(var i = 0; i < placeHolders.length; i++) {
 68 				if(!this.isAttachedPlaceHolder(placeHolders[i])) {
 69 					this.attachMacro(placeHolders[i]);
 70 				}
 71 			}
 72 		}
 73 	},
 74 		
 75 	/**
 76 	 * Updates all layers immediately. If there're invalid layers, detachs and removes them.
 77 	 */
 78 	updateLayers: function() {
 79 		if(this.xed.getCurrentEditMode() !== 'wysiwyg') return;
 80 		
 81 		for(var i = 0; i < this.layers.length; i++) {
 82 			var layer = this.layers[i];
 83 			if(layer.isValid()) {
 84 				layer.updatePosition();
 85 			} else {
 86 				layer.detach();
 87 				this.layers.splice(i, 1);
 88 			}
 89 		}
 90 	},
 91 
 92 	onValidatorPreprocessing: function(html) {
 93 		var p = xq.compilePattern("<(IFRAME|SCRIPT|OBJECT|EMBED)\\s+[^>]+(?:/>|>.*?</(?:IFRAME|SCRIPT|OBJECT|EMBED)>)", "img");
 94 		html.value = html.value.replace(p, function(str, tag) {
 95 			var macro = this.xed.macroFactory.createMacroFromHtml(str);
 96 			return macro ? macro.createPlaceHolderHtml() : "";
 97 		}.bind(this));
 98 	},
 99 	
100 	onValidatorAfterStringValidation: function(html) {
101 		var p1 = /<img\s+[^>]*class="xqlayer"\s+[^>]*\/>/mg;
102 		var p2 = /<img\s+[^>]*long[D|d]esc="(.+?)"\s+[^>]*\/>/m;
103 		
104 		html.value = html.value.replace(p1, function(img) {
105 			var def = JSON.parse(unescape(img.match(p2)[1]));
106 			var macro = this.xed.macroFactory.createMacroFromDefinition(def);
107 			return macro.createHtml();
108 		}.bind(this));
109 	}
110 });
