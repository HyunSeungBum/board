1 /**
  2  * @requires macro/Base.js

  4 xq.macro.JavascriptMacro = xq.Class(xq.macro.Base,
  5 	/**
  6 	 * Javascript macro
  7 	 * 
  8 	 * @name xq.macro.JavascriptMacro
  9 	 * @lends xq.macro.JavascriptMacro.prototype
 10 	 * @extends xq.macro.Base
 11 	 * @constructor
 12 	 */
 13 	{
 14 	initFromHtml: function() {
 15 		var p = xq.compilePattern("src=[\"'](.+?)[\"']", "img");
 16 		this.params.url = p.exec(this.html)[1];
 17 	},
 18 	initFromParams: function() {
 19 		if(!xq.macro.JavascriptMacro.isSafeScript(this.params.url)) throw "Unknown src";
 20 	},
 21 	createHtml: function() {return '<script type="text/javascript" src="' + this.params.url + '"></script>'},
 22 	
 23 	onLayerInitialzied: function(layer) {
 24 		layer.getDoc().write(this.createHtml());
 25 	}
 26 });
 27 
 28 xq.macro.JavascriptMacro.recognize = function(html) {
 29 	var p = xq.compilePattern("<SCRIPT\\s+[^>]*src=[\"']([^\"']+)[\"'][^>]*(?:/>|>.*?</(?:SCRIPT)>)", "img");
 30 	var m = p.exec(html);
 31 	if(!m || !m[1]) return false;
 32 	return this.isSafeScript(m[1]);
 33 }
 34 xq.macro.JavascriptMacro.isSafeScript = function(url) {
 35 	var safeSrcs = {
 36 		googleGadget: /http:\/\/gmodules\.com\/ig\/ifr\?/img
 37 	};
 38 	for(var id in safeSrcs) {
 39 		if(url.match(safeSrcs[id])) return true;
 40 	}
 41 	return false;
 42 }
