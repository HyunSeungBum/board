1 /**
  2  * @requires XQuared.js
  3  * @requires validator/W3.js
  4  */
  5 xq.validator.Webkit = xq.Class(xq.validator.W3,
  6 	/**
  7 	 * @name xq.validator.Webkit
  8 	 * @lends xq.validator.Webkit.prototype
  9 	 * @extends xq.validator.W3
 10 	 * @constructor
 11 	 */
 12 	{
 13 	validateDom: function(element) {
 14 		var rdom = xq.rdom.Base.createInstance();
 15 		rdom.setRoot(element);
 16 		this.removeDangerousElements(element);
 17 		rdom.removePlaceHoldersAndEmptyNodes(element);
 18 		this.validateAppleStyleTags(element);
 19 	},
 20 	
 21 	validateString: function(html) {
 22 		try {
 23 			html = this.addNbspToEmptyBlocks(html);
 24 			html = this.performFullValidation(html);
 25 			html = this.insertNewlineBetweenBlockElements(html);
 26 		} catch(ignored) {}
 27 		
 28 		return html;
 29 	},
 30 	
 31 	invalidateDom: function(element) {
 32 		this.invalidateAppleStyleTags(element);
 33 	},
 34 	
 35 	invalidateString: function(html) {
 36 		html = this.replaceTag(html, "strong", "b");
 37 		html = this.replaceTag(html, "em", "i");
 38 		html = this.removeComments(html);
 39 		html = this.replaceNbspToBr(html);
 40 		return html;
 41 	},
 42 	
 43 	validateAppleStyleTags: function(element) {
 44 		try {
 45 			var rdom = xq.rdom.Base.createInstance();
 46 			rdom.setRoot(element);
 47 			
 48 			var nodes = xq.getElementsByClassName(rdom.getRoot(), "Apple-style-span");
 49 			var holder = [];
 50 			
 51 			for(var i = 0; i < nodes.length; i++) {
 52 				var node = nodes[i];
 53 				if(node.style.fontStyle === "italic") {
 54 					// span -> em
 55 					node = rdom.replaceTag("em", node);
 56 					node.style.fontStyle = "";
 57 					holder.push({node:node});
 58 				} else if(node.style.fontWeight === "bold") {
 59 					// span -> strong
 60 					node = rdom.replaceTag("strong", node);
 61 					node.style.fontWeight = "";
 62 					holder.push({node:node});
 63 				} else if(node.style.textDecoration === "underline") {
 64 					// span -> em.underline
 65 					node = rdom.replaceTag("em", node);
 66 					node.style.textDecoration = "";
 67 					holder.push({node:node, className: 'underline'});
 68 				} else if(node.style.textDecoration === "line-through") {
 69 					// span -> span.strike
 70 					node.style.textDecoration = "";
 71 					holder.push({node:node, className: 'strike'});
 72 				} else if(node.style.verticalAlign === "super") {
 73 					// span -> sup
 74 					node = rdom.replaceTag("sup", node);
 75 					node.style.verticalAlign = "";
 76 					holder.push({node:node});
 77 				} else if(node.style.verticalAlign === "sub") {
 78 					// span -> sup
 79 					node = rdom.replaceTag("sub", node);
 80 					node.style.verticalAlign = "";
 81 					holder.push({node:node});
 82 				} else if(node.style.fontFamily) {
 83 					// span -> span font-family
 84 					holder.push({node:node});
 85 				}
 86 			}
 87 			
 88 			for (var j = 0; j < holder.length; j++){
 89 				if (holder[j].className) {
 90 					holder[j].node.className = holder[j].className;
 91 				} else {
 92 					holder[j].node.removeAttribute("class");
 93 				}
 94 			}
 95 			
 96 		} catch(e){
 97 		}
 98 	},
 99 	
100 	invalidateAppleStyleTags: function(element) {
101 		var rdom = xq.rdom.Base.createInstance();
102 		rdom.setRoot(element);
103 		
104 		var len;
105 		// span.strike -> span, span... -> span
106 		var spans = rdom.getRoot().getElementsByTagName("span");
107 		for(var i = 0; i < spans.length; i++) {
108 			var node = spans[i];
109 			if(node.className == "strike") {
110 				node.className = "Apple-style-span";
111 				node.style.textDecoration = "line-through";
112 			} else if(node.style.fontFamily) {
113 				node.className = "Apple-style-span";
114 			}
115 			// TODO: bg/fg/font-size
116 		}
117 
118 		// em -> span, em.underline -> span
119 		var ems = rdom.getRoot().getElementsByTagName("em");
120 		len = ems.length;
121 		for(var i = 0; i < len; i++) {
122 			var node = ems[0];
123 			node = rdom.replaceTag("span", node);
124 			if(node.className === "underline") {
125 				node.className = "Apple-style-span";
126 				node.style.textDecoration = "underline";
127 			} else {
128 				node.className = "Apple-style-span";
129 				node.style.fontStyle = "italic";
130 			}
131 		}
132 		
133 		// strong -> span
134 		var strongs = rdom.getRoot().getElementsByTagName("strong");
135 		len = strongs.length;
136 		
137 		for(var i = 0; i < len; i++) {
138 			var node = strongs[0];
139 			node = rdom.replaceTag("span", node);
140 			node.className = "Apple-style-span";
141 			node.style.fontWeight = "bold";
142 		}
143 		
144 		// sup -> span
145 		var sups = rdom.getRoot().getElementsByTagName("sup");
146 		len = sups.length;
147 		
148 		for(var i = 0; i < len; i++) {
149 			var node = sups[0];
150 			node = rdom.replaceTag("span", node);
151 			node.className = "Apple-style-span";
152 			node.style.verticalAlign = "super";
153 		}
154 		
155 		// sub -> span
156 		var subs = rdom.getRoot().getElementsByTagName("sub");
157 		len = subs.length;
158 		
159 		for(var i = 0; i < len; i++) {
160 			var node = subs[0];
161 			node = rdom.replaceTag("span", node);
162 			node.className = "Apple-style-span";
163 			node.style.verticalAlign = "sub";
164 		}
165 	}
166 });
