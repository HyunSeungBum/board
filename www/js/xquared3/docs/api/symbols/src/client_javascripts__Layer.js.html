1 /**
  2  * @requires XQuared.js
  3  * @requires Editor.js
  4  */
  5 xq.Layer = xq.Class(/** @lends xq.Layer.prototype */{
  6 	/**
  7      * @constructs
  8      *
  9      * @param {xq.Editor} editor editor instance
 10      * @param {Element} element designMode document's element. Layer instance will be attached to this element
 11      * @param {String} html HTML for body.
 12 	 */
 13 	initialize: function(editor, element, html) {
 14 		xq.addToFinalizeQueue(this);
 15 		
 16 		this.margin = 4;
 17 		this.editor = editor;
 18 		this.element = element;
 19 		this.frame = this.editor._createIFrame(this.editor.getOuterDoc(), this.element.offsetWidth - (this.margin * 2) + "px", this.element.offsetHeight + (this.margin * 2) + "px");
 20 		this.editor.getOuterDoc().body.appendChild(this.frame);
 21 		this.doc = editor._createDoc(
 22 			this.frame,
 23 			'<style type="text/css">html, body {margin:0px; padding:0px; background-color: transparent; width: 100%; height: 100%; overflow: hidden;}</style>',
 24 			[], [], null, null, html
 25 		);
 26 		this.frame.style.position = "absolute";
 27 		this.updatePosition();
 28 	},
 29 	
 30 	getFrame: function() {
 31 		return this.frame;
 32 	},
 33 	
 34 	getDoc: function() {
 35 		return this.doc;
 36 	},
 37 	
 38 	getBody: function() {
 39 		return this.doc.body;
 40 	},
 41 	
 42 	isValid: function() {
 43 		return this.element && this.element.parentNode && this.element.offsetParent;
 44 	},
 45 	
 46 	detach: function() {
 47 		this.frame.parentNode.removeChild(this.frame);
 48 		
 49 		this.frame = null;
 50 		this.element = null;
 51 	},
 52 	
 53 	updatePosition: function() {
 54 		// calculate element position
 55 		var offset = xq.getCumulativeOffset(this.element, this.editor.rdom.getRoot());
 56 		
 57 		// and scroll position
 58 		var doc = this.editor.getDoc();
 59 		var body = this.editor.getBody();
 60 		offset.left -= doc.documentElement.scrollLeft + body.scrollLeft - this.margin;
 61 		offset.top -= doc.documentElement.scrollTop + body.scrollTop - this.margin;
 62 		
 63 		// apply new position
 64 		this.frame.style.left = offset.left + "px";
 65 		this.frame.style.top = offset.top + "px";
 66 		
 67 		// perform autofit
 68 		var newWidth = this.doc.body.scrollWidth + (this.margin - 1) * 2;
 69 		var newHeight = this.doc.body.scrollHeight + (this.margin - 1) * 2;
 70 			
 71 		// without -1, the element increasing slowly. 
 72 		this.element.width = newWidth;
 73 		this.element.height = newHeight;
 74 		
 75 		// resize frame
 76 		this.frame.style.width = this.element.offsetWidth - (this.margin * 2) + "px";
 77 		this.frame.style.height = this.element.offsetHeight - (this.margin * 2) + "px";
 78 	}
 79 });
