105 	getAutocorrections: function() {
106 		return [
107 			{id:'bullet', criteria: /^(\s|\ \;)*(\*|-)(\s|\ \;).+$/, handler: function(xed, rdom, block, text) {
108 				rdom.pushMarker();
109 				rdom.removePlaceHoldersAndEmptyNodes(block);
110 				block.innerHTML = block.innerHTML.replace(/((\s| )*(\*|\-)\s*)/, "");
111 				if(block.nodeName === "LI") xed.handleIndent();
112 				if(block.parentNode.nodeName !== "UL") xed.handleList('UL');
113 				rdom.popMarker(true);
114 			}},
115 			{id:'numbering', criteria: /^(\s|\ \;)*(\d\.|#)(\s|\ \;).+$/, handler: function(xed, rdom, block, text) {
116 				rdom.pushMarker();
117 				rdom.removePlaceHoldersAndEmptyNodes(block);
118 				block.innerHTML = block.innerHTML.replace(/(\s| )*(\d\.|\#)\s*/, "")
119 				if(block.nodeName === "LI") xed.handleIndent();
120 				if(block.parentNode.nodeName !== "OL") xed.handleList('OL');
121 				rdom.popMarker(true);
122 			}},
123 			{id:'imageUrl', criteria: /https?:\/\/.*?\/(.*?\.(jpg|jpeg|gif|bmp|png))$/i, handler: function(xed, rdom, block, text) {
124 				var fileName = text.match(/https?:\/\/.*?\/(.*?\.(jpg|jpeg|gif|bmp|png))$/i)[1];
125 				block.innerHTML = "";
126 				var img = rdom.createElement("img");
127 				img.src = text;
128 				img.alt = fileName;
129 				img.title = fileName;
130 				block.appendChild(img);
131 				rdom.selectElement(block);
132 				rdom.collapseSelection(false);
133 			}},
134 			{id:'separator', criteria: /^---+(\ |\s)*$/, handler: function(xed, rdom, block, text) {
135 				if(rdom.tree.isBlockContainer(block)) block = rdom.wrapAllInlineOrTextNodesAs("P", block, true)[0];
136 				rdom.insertNodeAt(rdom.createElement("HR"), block, "before");
137 				block.innerHTML = "";
138 				rdom.placeCaretAtStartOf(block);
139 				return true;
140 			}},
141 			{id:'heading', criteria: /^\=+[^=]*\=+(\ |\s)*$/, handler: function(xed, rdom, block, text) {
142 				var textWithoutEqualMarks = text.strip().replace(/=/g, "");
143 				var level = Math.min(6, parseInt((text.length - textWithoutEqualMarks.length) / 2))
144 				xed.handleApplyBlock('H' + level);
145 				block = rdom.getCurrentBlockElement();
146 				block.innerHTML = textWithoutEqualMarks;
147 				rdom.selectElement(block);
148 				rdom.collapseSelection();
149 			}}
150 		];
151 	},

153 	getAutocompletions: function() {
154 		return [
155 			{
156 				id:'isbn',
157 				criteria: /@ISBN:\d+$/i,
158 				handler: function(xed, rdom, block, wrapper, text) {
159 					var isbn = text.split(":")[1]
160 					var korean = isbn.indexOf("97889") === 0 || isbn.indexOf("89") === 0
161 					var href = korean ?
162 						"http://www.aladdin.co.kr/shop/wproduct.aspx?ISBN=" :
163 						"http://www.amazon.com/exec/obidos/ISBN="
164 					var node = rdom.createElement('A');
165 					node.innerHTML = 'ISBN:' + isbn;
166 					node.href = href + isbn;
167 					node.className = 'external';
168 					node.title = 'ISBN:' + isbn;
169 					
170 					wrapper.innerHTML = "";
171 					wrapper.appendChild(node);
172 				}
173 			},
174 			{
175 				id:'anchor',
176 				criteria: /@A(:(.+))?$/i,
177 				handler: function(xed, rdom, block, wrapper, text) {
178 					var m = text.match(/@A(:(.+))?$/i);
179 					var anchorId = m[2] ? m[2] : function() {
180 						var id = 0;
181 						while(true) {
182 							var element = rdom.$("a" + (id));
183 							if(!element) return "a" + id;
184 							id++;
185 						}
186 					}();
187 					
188 					var node = rdom.createElement('A');
189 					node.id = anchorId;
190 					node.href = '#' + anchorId;
191 					node.className = 'anchor';
192 					node.title = 'Anchor ' + anchorId;
193 					node.innerHTML = '(' + anchorId + ')';
194 	
195 					wrapper.innerHTML = "";
196 					wrapper.appendChild(node);
197 				}
198 			}
199 		];
200 	},

211 						date: today.getDate(),
212 						hour: today.getHours(),
213 						min: today.getMinutes(),
214 						sec: today.getSeconds()
215 					};
216 					
217 					return html.replace(/\{xq:(year|month|date|hour|min|sec)\}/img, function(text, keyword) {

220 				}
