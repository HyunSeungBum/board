1 /**
  2  * @requires XQuared.js
  3  * @requires Browser.js
  4  * @requires Editor.js
  5  * @requires plugin/Base.js
  6  */
  7 xq.plugin.EditorResizePlugin = xq.Class(xq.plugin.Base,
  8 	/**
  9 	 * @name xq.plugin.EditorResizePlugin
 10 	 * @lends xq.plugin.EditorResizePlugin.prototype
 11 	 * @extends xq.plugin.Base
 12 	 * @constructor
 13 	 */
 14 	{
 15 	isEventListener: function() {return true;},
 16 	
 17 	onAfterLoad: function(xed) {
 18 		this.xed = xed;
 19 		this.bar = null;
 20 		this.screen = null;
 21 		this.active = false;
 22 	},
 23 	
 24 	onEditorInitialized: function(xed) {
 25 		xed.registerEventFirer("Editor", "Resized");
 26 		
 27 		var wrapper = this.xed.getOutmostWrapper();
 28 		var doc = wrapper.ownerDocument;
 29 		
 30 		// create resize bar
 31 		this.bar = doc.createElement("DIV");
 32 		if(xq.Browser.isIE6) this.bar.innerHTML = "<span></span>";
 33 		this.bar.style.height = "10px";
 34 		this.bar.style.backgroundImage = 'url(../images/plugin/resize_arrow.gif)';
 35 		this.bar.style.backgroundRepeat = 'no-repeat';
 36 		this.bar.style.backgroundPosition = 'center';
 37 		this.bar.style.cursor = "n-resize";
 38 		
 39 		wrapper.appendChild(this.bar);
 40 		
 41 		// register event
 42 		xq.observe(this.bar, 'mousedown', this.onMousedown.bindAsEventListener(this));
 43 		xq.observe(document, 'mouseup', this.onMouseup.bindAsEventListener(this));
 44 		xq.observe(this.bar, 'click', this.onMouseup.bindAsEventListener(this));
 45 		this.mousemoveHandler = this.onMousemove.bindAsEventListener(this);
 46 	},
 47 	
 48 	onMousedown: function(e) {
 49 		if(this.active) return;
 50 		
 51 		xq.observe(document, 'mousemove', this.mousemoveHandler);
 52 		this.last = e.screenY;
 53 
 54 		var wrapper = this.xed.getOutmostWrapper();
 55 		var doc = wrapper.ownerDocument;
 56 		var wysiwygDiv = this.xed.getWysiwygEditorDiv();
 57 		var sourceDiv = this.xed.getSourceEditorDiv();
 58 		var visibleDiv = this.xed.getCurrentEditMode() == "wysiwyg" ? wysiwygDiv : sourceDiv;
 59 		var location = xq.getCumulativeOffset(visibleDiv);
 60 		
 61 		// create screen
 62 		this.screen = doc.createElement("DIV");
 63 		if(xq.Browser.isIE6) this.screen.innerHTML = "<span></span>";
 64 		
 65 		if(xq.Browser.isIE6) {
 66 			this.screen.style.backgroundColor = "#EEE";
 67 			wysiwygDiv.style.display = "none";
 68 		} else {
 69 			this.screen.style.position = "absolute";
 70 			this.screen.style.left = location.left + "px";
 71 			this.screen.style.top = location.top + "px";
 72 		}
 73 		
 74 		this.screen.style.width = visibleDiv.clientWidth + "px";
 75 		this.screen.style.height = visibleDiv.clientHeight + "px";
 76 		wrapper.insertBefore(this.screen, visibleDiv);
 77 		
 78 		this.resize(e.screenY);
 79 		this.active = true;
 80 		
 81 		xq.stopEvent(e);
 82 		return true;
 83 	},
 84 	onMouseup: function(e) {
 85 		if(!this.active) return;
 86 		this.active = false;
 87 		
 88 		xq.stopObserving(document, 'mousemove', this.mousemoveHandler);
 89 		this.resize(e.screenY);
 90 		
 91 		if(xq.Browser.isIE6) {
 92 			var wysiwygDiv = this.xed.getWysiwygEditorDiv();
 93 			var sourceDiv = this.xed.getSourceEditorDiv();
 94 			var visibleDiv = this.xed.getCurrentEditMode() == "wysiwyg" ? wysiwygDiv : sourceDiv;
 95 			visibleDiv.style.display = "block";
 96 		}
 97 		
 98 		this.screen.parentNode.removeChild(this.screen);
 99 		this.screen = null;
100 
101 		this.xed._fireOnResized(this.xed);
102 		
103 		xq.stopEvent(e);
104 		return true;
105 	},
106 	onMousemove: function(e) {
107 		this.resize(e.screenY);
108 
109 		xq.stopEvent(e);
110 		return true;
111 	},
112 	resize: function(y) {
113 		var delta = y - this.last;
114 		
115 		var wysiwygDiv = this.xed.getWysiwygEditorDiv();
116 		var sourceDiv = this.xed.getSourceEditorDiv();
117 		var newHeight = Math.max(0, this.screen.clientHeight + delta);
118 		
119 		sourceDiv.style.height = wysiwygDiv.style.height = this.screen.style.height = newHeight + "px";
120 		
121 		this.last = y;
122 	}
123 });
