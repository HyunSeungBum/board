1 /**
  2  * @requires macro/Base.js
  3  */
  4 xq.macro.IFrameMacro = xq.Class(xq.macro.Base,
  5 	/**
  6 	 * IFrame macro
  7 	 * 
  8 	 * @name xq.macro.IFrameMacro
  9 	 * @lends xq.macro.IFrameMacro.prototype
 10 	 * @extends xq.macro.Base
 11 	 * @constructor
 12 	 */
 13 	{
 14 	initFromHtml: function() {
 15 		this.params.html = this.html;
 16 	},
 17 	initFromParams: function() {
 18 		if(this.params.html) return;
 19 		
 20 		var sb = [];
 21 		sb.push('<iframe');
 22 		for(var attrName in this.params) {
 23 			var attrValue = this.params[attrName];
 24 			if(attrValue) sb.push(' ' + attrName.substring("p_".length) + '="' + attrValue + '"');
 25 		}
 26 		sb.push('></iframe>');
 27 		this.params = {html:sb.join("")};
 28 	},
 29 	createHtml: function() {
 30 		return this.params.html;
 31 	}
 32 });
 33 xq.macro.IFrameMacro.recognize = function(html) {
 34 	var p = xq.compilePattern("<IFRAME\\s+[^>]+(?:/>|>.*?</(?:IFRAME)>)", "img");
 35 	return !!html.match(p);
 36 }
