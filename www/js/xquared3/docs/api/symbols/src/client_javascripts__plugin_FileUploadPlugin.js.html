 62 		xed.handleFileUpload = function(isSingleFileUpload) {
 63 			var requiredMajorVersion = 9;
 64 			var requiredMinorVersion = 0;
 65 			var requiredRevision = 0;
 66 
 67 			this.isSingleFileUpload = isSingleFileUpload || !DetectFlashVer(requiredMajorVersion, requiredMinorVersion, requiredRevision);
 68 			if (this.fileUploadController.dialog) this.fileUploadController.dialog.close();
 69 
 70 			var dialog = new xq.ui.FormDialog(
 71 				this,
 72 				(this.isSingleFileUpload)? xq.ui_templates.basicFileUploadDialog : xq.ui_templates.basicMultiFileUploadDialog,
 73 				function(dialog) {
 74 					xed.fileUploadController.load(this);
 75 				},
 76 				function(dialog) {
 77 					xed.focus();
 78 
 79 					if(xq.Browser.isTrident) {
 80 						var rng = xed.rdom.rng();
 81 						rng.moveToBookmark(bm);
 82 						rng.select();
 83 					}
 84 					
 85 					// cancel?
 86 					if(!dialog) {
 87 						xed.fileUploadController.dialog = null;
 88 						if (!xed.isSingleFileUpload && xed.fileUploadController.swfUploader){
 89 							xed.fileUploadController.swfUploader.destroy();
 90 							xed.fileUploadController.swfUploader = null;
 91 						}
 92 						return;
 93 					}
 94 
 95 					if (xed.isSingleFileUpload){
 96 						this.form.submit();
 97 					} else {
 98 						xed.fileUploadController.startUpload();
 99 					}
100 					
101 					return true;
102 				}
103 			);
104 
105 			this.fileUploadController.dialog = dialog;
106 			if(xq.Browser.isTrident) var bm = this.rdom.rng().getBookmark();
107 			dialog.show({position: 'centerOfEditor', notSelfClose: true});
108 
109 			return true;
110 		}

112 		xed.fileUploadListener = {
113 			fileListContainer: null,
114 			isError: false,
115 			onReady: function(){
116 				
117 			},
118 			onProgress: function(file, bytesLoaded, bytesTotal){
119 				var nPerc = bytesLoaded / bytesTotal * 100;
120 				xq.$('file-' + file.id + '-progress').style.width = nPerc + '%';
121 			},
122 			onSelectStart: function(){
123 				var ul = document.createElement('UL');
124 				ul.id = 'uploadFileList';
125 				this.fileListContainer = ul;
126 			},
127 			onSelectComplete: function(selectedFiles, successFiles, totalFiles){
128 				if(!successFiles) {
129 					this.filListContainer = null;
130 				} else {
131 					var fileListContainer = xq.$('fileListContainer');
132 					var fileUploadContainer = xq.$('fileUploadContainer');
133 	
134 				  	fileListContainer.style.display = 'block';
135 					
136 				  	if (fileUploadContainer.style.width != '1px'){
137 					  	fileUploadContainer.style.width = '1px';
138 					  	fileUploadContainer.style.height = '1px';
139 					  	fileUploadContainer.style.border = 'none';
140 					  	fileUploadContainer.style.padding = 0;
141 					  	fileUploadContainer.style.backgroundColor = '#ffffff';
142 	
143 					  	xed.fileUploadController.swfUploader.getMovieElement().style.width = '1px';
144 					  	xed.fileUploadController.swfUploader.getMovieElement().style.height = '1px';
145 	
146 					  	fileUploadContainer.removeChild(fileUploadContainer.getElementsByTagName('P')[0]);
147 				  	}
148 					
149 				  	fileListContainer.innerHTML = "";
150 				  	fileListContainer.appendChild(this.fileListContainer);
151 				}
152 			},
153 			
154 			onSelect: function(file){
155 				var li = document.createElement("LI");
156 				li.id = "file_" + file.id;
157 
158 				var fileName = file["name"];
159 				var charlength = fileName.length;
160 				var maxLength = 30;
161 				
162 				for (var j = 0; j < fileName.length; j++){
163 					if (fileName.substr(j,1).charCodeAt() >= 256) {
164 						charlength++;
165 						maxLength--;
166 					}
167 				}
168 				if (charlength > maxLength) fileName = fileName.substr(0, maxLength) + '..';
169 
170 				var divLeft = document.createElement("DIV");
171 				divLeft.innerHTML = fileName;
172 				divLeft.className = 'div_left';
173 				li.appendChild(divLeft);
174 
175 				var span = document.createElement("SPAN");
176 				divLeft.appendChild(span);
177 				span.className = 'capacity';
178 				
179 				var sizes = ['B', 'KB', 'MB', 'GB'];
180 				var fileSize = file["size"];
181 				var i = 0;
182 				while (fileSize >= 1024 && i + 1 < sizes.length) {
183 				    i++;
184 				    fileSize = fileSize / 1024;
185 				}
186 				
187 				span.innerHTML = '(' + Math.ceil(fileSize) + sizes[i] + ')';
188 
189 				var div = document.createElement("DIV");
190 
191 				li.appendChild(div);
192 
193 				div.id = 'file-' + file.id + '-option';
194 				div.className = 'cancel_button';
195 
196 				div.innerHTML = '<a href="#" onclick="xed.fileUploadController.cancelUpload(\'' + file.id + '\'); return false;">Cancel</a>'
197 
198 				var graphdiv = document.createElement("DIV");
199 				li.appendChild(graphdiv);
200 				graphdiv.className = 'graph';
201 
202 				var progressDiv = document.createElement('DIV');
203 				graphdiv.appendChild(progressDiv);
204 				progressDiv.id = 'file-' + file.id + '-progress';
205 
206 				var img = document.createElement('IMG');
207 				img.src = xed.config.imagePathForDialog + 'multifile_fill.gif';
208 				img.alt = 'Upload Complete';
209 				
210 				progressDiv.appendChild(img);
211 				graphdiv.style.display = 'none';
212 				
213 				this.fileListContainer.appendChild(li);
214 			},
215 			onUploadStart: function(file){
216 				xq.$('file-' + file.id + '-progress').parentNode.style.display = 'block'
217 				xq.$('file-' + file.id + '-option').style.display = 'none'
218 			},
219 			onSuccess: function(file, serverData){
220 				if (xed.isSingleFileUpload){
221 					var doc = xq.$("uploadTarget");
222 					var serverData = doc.contentWindow.document.body.innerHTML;
223 				}
224 				if (!serverData) return;
225 				eval("var data = " + serverData);
226 					
227 				if(data && data.success) {
228 					if (xed.isSingleFileUpload){
229 						xed.insertImageFileToEditor(data.file_url, file.name, window.parent.xed);
230 						this.onQueueComplete();
231 					} else {
232 						xed.insertImageFileToEditor(data.file_url, file.name);
233 						xq.$('file-' + file.id + '-progress').style.width = '100%';
234 					}
235 				} else {
236 					alert(data.message);
237 				}
238 				
239 			},
240 			onSelectError : function (file, errorCode, message) {
241 				try {
242 					var errorName = "";
243 					switch (errorCode) {
244 					case SWFUpload.QUEUE_ERROR.QUEUE_LIMIT_EXCEEDED:
245 						errorName = "QUEUE LIMIT EXCEEDED";
246 						break;
247 					case SWFUpload.QUEUE_ERROR.FILE_EXCEEDS_SIZE_LIMIT:
248 						errorName = "FILE EXCEEDS SIZE LIMIT";
249 						break;
250 					case SWFUpload.QUEUE_ERROR.ZERO_BYTE_FILE:
251 						errorName = "ZERO BYTE FILE";
252 						break;
253 					case SWFUpload.QUEUE_ERROR.INVALID_FILETYPE:
254 						errorName = "INVALID FILE TYPE";
255 						break;
256 					default:
257 						errorName = "UNKNOWN";
258 						break;
259 					}
260 		
261 				} catch (ex) {
262 				}
263 			},
264 			onQueueComplete: function(numFilesUploaded){
265 				if (!this.isError) {
266 					if (!xed.isSingleFileUpload && xed.fileUploadController.swfUploader){
267 						xed.fileUploadController.swfUploader.destroy();
268 						xed.fileUploadController.swfUploader = null;
269 					}
270 					xed.fileUploadController.dialog.close();
271 					xed.fileUploadController.dialog = null;
272 				}
273 			},
274 			onUploadError: function(file, errorCode, message){
275 				this.isError = true;
276 				try {
277 					var errorName = "";
278 					switch (errorCode) {
279 					case SWFUpload.UPLOAD_ERROR.HTTP_ERROR:
280 						errorName = "HTTP ERROR";
281 						break;
282 					case SWFUpload.UPLOAD_ERROR.MISSING_UPLOAD_URL:
283 						errorName = "MISSING UPLOAD URL";
284 						break;
285 					case SWFUpload.UPLOAD_ERROR.IO_ERROR:
286 						errorName = "IO ERROR";
287 						break;
288 					case SWFUpload.UPLOAD_ERROR.SECURITY_ERROR:
289 						errorName = "SECURITY ERROR";
290 						break;
291 					case SWFUpload.UPLOAD_ERROR.UPLOAD_LIMIT_EXCEEDED:
292 						errorName = "UPLOAD LIMIT EXCEEDED";
293 						break;
294 					case SWFUpload.UPLOAD_ERROR.UPLOAD_FAILED:
295 						errorName = "UPLOAD FAILED";
296 						break;
297 					case SWFUpload.UPLOAD_ERROR.SPECIFIED_FILE_ID_NOT_FOUND:
298 						errorName = "SPECIFIED FILE ID NOT FOUND";
299 						break;
300 					case SWFUpload.UPLOAD_ERROR.FILE_VALIDATION_FAILED:
301 						errorName = "FILE VALIDATION FAILED";
302 						break;
303 					case SWFUpload.UPLOAD_ERROR.FILE_CANCELLED:
304 						errorName = "FILE CANCELLED";
305 						break;
306 					case SWFUpload.UPLOAD_ERROR.UPLOAD_STOPPED:
307 						errorName = "FILE STOPPED";
308 						break;
309 					default:
310 						errorName = "UNKNOWN";
311 						break;
312 					}
313 					
314 					var index = file.id;
315 					var div = xq.$('file-' + index + '-option');
316 					if (!div) return ;
317 					div.style.display = "block"
318 					div.innerHTML = "";
319 					if (xq.$('file-' + index + '-progress')) xq.$('file-' + index + '-progress').parentNode.style.display ="none";
320 					var sError = document.createElement("SPAN");
321 					sError.className = 'error font-variation';
322 					sError.innerHTML = (errorName)? errorName:'Error';
323 					div.appendChild(sError);
324 					
325 				} catch (ex) {
326 				}
327 				
328 			}
329 		}
