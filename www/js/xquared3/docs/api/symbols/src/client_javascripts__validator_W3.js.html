1 /**

  5 xq.validator.W3 = xq.Class(xq.validator.Base,
  6 	/**
  7 	 * @name xq.validator.W3
  8 	 * @lends xq.validator.W3.prototype
  9 	 * @extends xq.validator.Base
 10 	 * @constructor
 11 	 */
 12 	{
 13 	validateDom: function(element) {
 14 		var rdom = xq.rdom.Base.createInstance();
 15 		rdom.setRoot(element);
 16 		this.removeDangerousElements(element);
 17 		rdom.removePlaceHoldersAndEmptyNodes(element);
 18 		this.validateFont(element);
 19 	},
 20 	
 21 	validateString: function(html) {
 22 		try {
 23 			html = this.replaceTag(html, "b", "strong");
 24 			html = this.replaceTag(html, "i", "em");
 25 			
 26 			html = this.removeComments(html);
 27 			html = this.validateStrike(html);
 28 			html = this.validateUnderline(html);
 29 			html = this.addNbspToEmptyBlocks(html);
 30 			html = this.performFullValidation(html);
 31 			html = this.insertNewlineBetweenBlockElements(html);
 32 			
 33 		} catch(ignored) {}
 34 		
 35 		return html;
 36 	},
 37 	
 38 	invalidateDom: function(element) {
 39 		this.invalidateFont(element);
 40 		this.invalidateStrikesAndUnderlines(element);
 41 	},
 42 	
 43 	invalidateString: function(html) {
 44 		html = this.replaceTag(html, "strong", "b");
 45 		html = this.replaceTag(html, "em", "i");
 46 		html = this.removeComments(html);
 47 		html = this.replaceNbspToBr(html);
 48 		return html;
 49 	},
 50 	
 51 	performFullValidation: function(html) {
 52 		html = this.validateSelfClosingTags(html);
 53 		html = this.applyWhitelist(html);
 54 		
 55 		if(this.urlValidationMode === 'relative') {
 56 			html = this.makeUrlsRelative(html);
 57 		} else if(this.urlValidationMode === 'host_relative') {
 58 			html = this.makeUrlsHostRelative(html);
 59 		} else if(this.urlValidationMode === 'absolute') {
 60 			html = this.makeUrlsAbsolute(html);
 61 		}
 62 
 63 		return html;
 64 	},
 65 	
 66 	insertNewlineBetweenBlockElements: function(html) {
 67 		var blocks = new xq.DomTree().getBlockTags().join("|");
 68 		var regex = new RegExp("</(" + blocks + ")>([^\n])", "img");
 69 		return html.replace(regex, '</$1>\n$2');
 70 	},
 71 	
 72 	replaceNbspToBr: function(content) {
 73 		var blocks = new xq.DomTree().getBlockTags().join("|");
 74 		
 75 		// Safari replaces   into \xA0
 76 		var regex = new RegExp("<(" + blocks + ")([^>]+?)?>( |\xA0)?</(" + blocks + ")>", "img");
 77 		var rdom = xq.rdom.Base.createInstance();
 78 		return content.replace(regex, '<$1$2>' + rdom.makePlaceHolderString() + '</$4>');
 79 	}
 80 });
