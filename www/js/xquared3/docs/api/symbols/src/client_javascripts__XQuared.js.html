 14 /**
 15  * Compiles regular expression pattern if possible.
 16  *
 17  * @param {String} p Regular expression.
 18  * @param {String} f Flags.
 19  */
 20 xq.compilePattern = function(p, f) {
 21 	if(!RegExp.prototype.compile) return new RegExp(p, f);
 22 	
 23 	var r = new RegExp();
 24 	r.compile(p, f);
 25 	return r;
 26 }

 28 /**
 29  * @class Simple class based OOP framework
 30  */
 31 xq.Class = function() {
 32 	var parent = null, properties = xq.$A(arguments), key;
 33 	if (typeof properties[0] === "function") {
 34 		parent = properties.shift();
 35 	}
 36 	
 37 	function klass() {
 38 		this.initialize.apply(this, arguments);
 39 	}
 40 	
 41 	if(parent) {
 42 		for (key in parent.prototype) {
 43 			klass.prototype[key] = parent.prototype[key];
 44 		}
 45 	}
 46 	
 47 	for (key in properties[0]) if(properties[0].hasOwnProperty(key)){
 48 		klass.prototype[key] = properties[0][key];
 49 	}
 50 	
 51 	if (!klass.prototype.initialize) {
 52 		klass.prototype.initialize = function() {};
 53 	}
 54 	
 55 	klass.prototype.constructor = klass;
 56 	
 57 	return klass;
 58 };

 96  * Stops event propagation.

 98  * @param {Event} e Event to stop.

100 xq.stopEvent = function(e, returnValue) {
101       if(e.preventDefault) {
102     	  e.preventDefault();
103       }
104       if(e.stopPropagation) {
105     	  e.stopPropagation();
106       }
107       
108       if(returnValue)
109       {
110 		e.returnValue = returnValue;
111       }
112       else
113       {
114       	e.returnValue = false;
115       }
116       
117       e.cancelBubble = true;
118       e.stopped = true;
119 };

185 xq.serializeForm = function(f) {
186 	var options = {hash: true};
187 	var data = {};
188 	var elements = f.getElementsByTagName("*");
189 	for(var i = 0; i < elements.length; i++) {
190 		var element = elements[i];
191 		var tagName = element.tagName.toLowerCase();
192 		if(element.disabled || !element.name || ['input', 'textarea', 'option', 'select'].indexOf(tagName) === -1) {
193 			continue;
194 		}
195 		
196 		var key = element.name;
197 		var value = xq.getValueOfElement(element);
198 		
199 		if(value === undefined) {
200 			continue;
201 		}
202 		
203 		if(key in data) {
204 			if(data[key].constructor === Array) {
205 				data[key] = [data[key]];
206 			}
207 			data[key].push(value);
208 		} else {
209 			data[key] = value;
210 		}
211 	}
212 	return data;
213 };

356 /**
357  * Make given object as event source
358  *
359  * @param {Object} object target object
360  * @param {String} prefix prefix for generated functions
361  * @param {Array} events array of string which contains name of events
362  */
363 xq.asEventSource = function(object, prefix, events) {
364 	object.autoRegisteredEventListeners = [];
365 	object.registerEventFirer = function(prefix, name) {
366 		this["_fireOn" + name] = function() {
367 			for(var i = 0; i < this.autoRegisteredEventListeners.length; i++) {
368 				var listener = this.autoRegisteredEventListeners[i];
369 				var func = listener["on" + prefix + name];
370 				if(func) {
371 					func.apply(listener, xq.$A(arguments));
372 				}
373 			}
374 		};
375 	};
376 	object.addListener = function(l) {
377 		this.autoRegisteredEventListeners.push(l);
378 	};
379 	
380 	for(var i = 0; i < events.length; i++) {
381 		object.registerEventFirer(prefix, events[i]);
382 	}
383 };

396 /**
397  * Element to JSON mapper
398  */
399 xq.element2json = function(element) {
400 	var o, i, childElements;
401 	
402 	if(element.nodeName === 'DL') {
403 		o = {};
404 		childElements = xq.findChildElements(element);
405 		for(i = 0; i < childElements.length; i++) {
406 			var dt = childElements[i];
407 			var dd = childElements[++i];
408 			o[dt.innerHTML] = xq.element2json(xq.findChildElements(dd)[0]);
409 		}
410 		return o;
411 	} else if (element.nodeName === 'OL') {
412 		o = [];
413 		childElements = xq.findChildElements(element);
414 		for(i = 0; i < childElements.length; i++) {
415 			var li = childElements[i];
416 			o[i] = xq.element2json(xq.findChildElements(li)[0]);
417 		}
418 	} else if(element.nodeName === 'SPAN' && element.className === 'number') {
419 		return parseFloat(element.innerHTML);
420 	} else if(element.nodeName === 'SPAN' && element.className === 'string') {
421 		return element.innerHTML;
422 	} else { // ignore textnode or unknown tag
423 		return null;
424 	}
425 };

436 xq._json2html = function(o, sb) {
437 	if(typeof o === 'number') {
438 		sb.push('<span class="number">' + o + '</span>');
439 	} else if(typeof o === 'string') {
440 		sb.push('<span class="string">' + o.escapeHTML() + '</span>');
441 	} else if(o.constructor === Array) {
442 		sb.push('<ol>');
443 		for(var i = 0; i < o.length; i++) {
444 			sb.push('<li>');
445 			xq._json2html(o[i], sb);
446 			sb.push('</li>');
447 		}
448 		sb.push('</ol>');
449 	} else { // Object
450 		sb.push('<dl>');
451 		for (var key in o) if (o.hasOwnProperty(key)) {
452 			sb.push('<dt>' + key + '</dt>');
453 			sb.push('<dd>');
454 			xq._json2html(o[key], sb);
455 			sb.push('</dd>');
456 		}
457 		sb.push('</dl>');
458 	}
459 };

707 xq.loadOthers = function() {
708 	var basePath = xq.getXquaredScriptBasePath();
709 	var others = xq.getXquaredScriptFileNames();
710 	
711 	// Xquared.js(this file) should not be loaded again. So the value of "i" starts with 1 instead of 0
712 	for(var i = 1; i < others.length; i++) {
713 		xq.loadScript(basePath + others[i]);
714 	}
715 };
