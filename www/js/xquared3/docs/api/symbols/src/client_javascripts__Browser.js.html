1 /**
  2  * @namespace Contains browser detection codes
  3  * 
  4  * @requires XQuared.js
  5  */
  6 xq.Browser = new function() {
  7 	
  8 	this.language = (navigator.appName == 'Netscape') ? navigator.language : navigator.browserLanguage;
  9 	
 10 	// By Rendering Engines
 11 	
 12 	/** 
 13 	 * True if rendering engine is Trident
 14 	 * @type boolean
 15 	 */
 16 	this.isTrident = navigator.appName === "Microsoft Internet Explorer",
 17 	
 18 	/**
 19 	 * True if rendering engine is Webkit
 20 	 * @type boolean
 21 	 */
 22 	this.isWebkit = navigator.userAgent.indexOf('AppleWebKit/') > -1,
 23 	
 24 	/**
 25 	 * True if rendering engine is Gecko
 26 	 * @type boolean
 27 	 */
 28 	this.isGecko = navigator.userAgent.indexOf('Gecko') > -1 && navigator.userAgent.indexOf('KHTML') === -1,
 29 	
 30 	/**
 31 	 * True if rendering engine is KHTML
 32 	 * @type boolean
 33 	 */
 34 	this.isKHTML = navigator.userAgent.indexOf('KHTML') !== -1,
 35 	
 36 	/**
 37 	 * True if rendering engine is Presto
 38 	 * @type boolean
 39 	 */
 40 	this.isPresto = navigator.appName === "Opera",
 41 	
 42 	
 43 	
 44 	// By Platforms
 45 	/**
 46 	 * True if platform is Mac
 47 	 * @type boolean
 48 	 */
 49 	this.isMac = navigator.userAgent.indexOf("Macintosh") !== -1,
 50 	
 51 	/**
 52 	 * True if platform is Ubuntu Linux
 53 	 * @type boolean
 54 	 */
 55 	this.isUbuntu = navigator.userAgent.indexOf('Ubuntu') !== -1,
 56 	
 57 	/**
 58 	 * True if platform is Windows
 59 	 * @type boolean
 60 	 */
 61 	this.isWin = navigator.userAgent.indexOf('Windows') !== -1,
 62 
 63 
 64 
 65 	// By Browsers
 66 	/**
 67 	 * True if browser is Internet Explorer
 68 	 * @type boolean
 69 	 */
 70 	this.isIE = navigator.appName === "Microsoft Internet Explorer",
 71 	
 72 	/**
 73 	 * True if browser is Internet Explorer 6
 74 	 * @type boolean
 75 	 */
 76 	this.isIE6 = navigator.userAgent.indexOf('MSIE 6') !== -1,
 77 	
 78 	/**
 79 	 * True if browser is Internet Explorer 7
 80 	 * @type boolean
 81 	 */
 82 	this.isIE7 = navigator.userAgent.indexOf('MSIE 7') !== -1,
 83 	
 84 	/**
 85 	 * True if browser is Internet Explorer 8
 86 	 * @type boolean
 87 	 */
 88 	this.isIE8 = navigator.userAgent.indexOf('MSIE 8') !== -1,
 89 	
 90 	/**
 91 	 * True if browser is Firefox
 92 	 * @type boolean
 93 	 */
 94 	this.isFF = navigator.userAgent.indexOf('Firefox') !== -1,
 95 	
 96 	/**
 97 	 * True if browser is Firefox 2
 98 	 * @type boolean
 99 	 */
100 	this.isFF2 = navigator.userAgent.indexOf('Firefox/2') !== -1,
101 	
102 	/**
103 	 * True if browser is Firefox 3
104 	 * @type boolean
105 	 */
106 	this.isFF3 = navigator.userAgent.indexOf('Firefox/3') !== -1,
107 	
108 	/**
109 	 * True if browser is Safari
110 	 * @type boolean
111 	 */
112 	this.isSafari = navigator.userAgent.indexOf('Safari') !== -1
