1 /**
  2  * Creates and returns instance of browser specific implementation.
  3  * 
  4  * @requires XQuared.js
  5  * @requires validator/Base.js
  6  * @requires validator/Trident.js
  7  * @requires validator/Gecko.js
  8  * @requires validator/Webkit.js
  9  */
 10 xq.validator.Base.createInstance = function(curUrl, urlValidationMode, whitelist) {
 11 	if(xq.Browser.isTrident) {
 12 		return new xq.validator.Trident(curUrl, urlValidationMode, whitelist);
 13 	} else if(xq.Browser.isWebkit) {
 14 		return new xq.validator.Webkit(curUrl, urlValidationMode, whitelist);
 15 	} else {
 16 		return new xq.validator.Gecko(curUrl, urlValidationMode, whitelist);
 17 	}
 18 }
