 17 	collectCells: function(cell){
 18 		var cells = [];
 19 		var x = this.getXIndexOf(cell);
 20 		var y = 0;
 21 		while(true) {
 22 			var cur = this.getCellAt(x, y);
 23 			if(!cur) break;
 24 			cells.push(cur);
 25 			y++;
 26 		}
 27 		
 28 		return cells
 29 	},

 31 		var row = this.rdom.createElement("TR");
 32 		var cells = tr.cells;

 34 			var cell = this.rdom.createElement(cells[i].nodeName);
 35 			this.rdom.correctEmptyElement(cell);
 36 			row.appendChild(cell);
 37 		}
 38 		return this.rdom.insertNodeAt(row, tr, where);
 39 	},

240 xq.RichTable.create = function(rdom, attrs) {
241 	if(["t", "tl", "lt"].indexOf(attrs.headerPositions) !== -1) var headingAtTop = true
242 	if(["l", "tl", "lt"].indexOf(attrs.headerPositions) !== -1) var headingAtLeft = true
243 
244 	var sb = []
245 	sb.push('<table class="datatable2" style="width:100%;">')
246 	
247 	// thead
248 	if(headingAtTop) {
249 		sb.push('<thead><tr>')
250 		for(var i = 0; i < attrs.cols; i++) sb.push('<th></th>')
251 		sb.push('</tr></thead>')
252 		attrs.rows -= 1
253 	}
254 		
255 	// tbody
256 	sb.push('<tbody>')
257 	for(var i = 0; i < attrs.rows; i++) {
258 		sb.push('<tr>')
259 		
260 		for(var j = 0; j < attrs.cols; j++) {
261 			if(headingAtLeft && j === 0) {
262 				sb.push('<th></th>')
263 			} else {
264 				sb.push('<td></td>')
265 			}
266 		}
267 		
268 		sb.push('</tr>')
269 	}
270 	sb.push('</tbody>')
271 	
272 	sb.push('</table>')
273 	
274 	// create DOM element
275 	var container = rdom.createElement("div");
276 	container.innerHTML = sb.join("");
277 	
278 	// correct empty cells and return
279 	var rtable = new xq.RichTable(rdom, container.firstChild);
280 	rtable.correctEmptyCells();
281 	return rtable;
