1 /**
  2  * @requires macro/Base.js

  4 xq.macro.FlashMovieMacro = xq.Class(xq.macro.Base,
  5 	/**
  6 	 * Flash movie macro
  7 	 * 
  8 	 * @name xq.macro.FlashMovieMacro
  9 	 * @lends xq.macro.FlashMovieMacro.prototype
 10 	 * @extends xq.macro.Base
 11 	 * @constructor
 12 	 */
 13 	{
 14 	initFromHtml: function() {
 15 		this.params.html = this.html;
 16 	},
 17 	initFromParams: function() {
 18 		if(xq.macro.FlashMovieMacro.skipProviderValidation) return true;
 19 		if(!xq.macro.FlashMovieMacro.recognize(this.params.html)) throw "Unknown src";
 20 	},
 21 	createHtml: function() {
 22 		return this.params.html;
 23 	}
 24 });

 49 xq.macro.FlashMovieMacro.recognize = function(html) {
 50 	if(xq.macro.FlashMovieMacro.skipProviderValidation) return true;
 51 	
 52 	var providers = xq.macro.FlashMovieMacro.whiteList;
 53 	
 54 	for(var id in providers) {
 55 		if(html.match(providers[id])) return true;
 56 	}
 57 	
 58 	return false;
 59 }
