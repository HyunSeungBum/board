1 /**
  2  * @requires XQuared.js
  3  * @requires validator/Base.js
  4  */
  5 xq.validator.Trident = xq.Class(xq.validator.Base,
  6 	/**
  7 	 * @name xq.validator.Trident
  8 	 * @lends xq.validator.Trident.prototype
  9 	 * @extends xq.validator.Base
 10 	 * @constructor
 11 	 */
 12 	{
 13 	validateDom: function(element) {
 14 		this.removeDangerousElements(element);
 15 		this.validateFont(element);
 16 	},
 17 	
 18 	validateString: function(html) {
 19 		try {
 20 			html = this.addNbspToEmptyBlocks(html);
 21 			html = this.validateStrike(html);
 22 			html = this.validateUnderline(html);
 23 			html = this.performFullValidation(html);
 24 		} catch(ignored) {}
 25 		
 26 		return html;
 27 	},
 28 	
 29 	invalidateDom: function(element) {
 30 		this.invalidateFont(element);
 31 		this.invalidateStrikesAndUnderlines(element);
 32 	},
 33 	
 34 	invalidateString: function(html) {
 35 		html = this.removeComments(html);
 36 		return html;
 37 	},
 38 	
 39 	performFullValidation: function(html) {
 40 		html = this.lowerTagNamesAndUniformizeQuotation(html);
 41 		html = this.validateSelfClosingTags(html);
 42 		html = this.applyWhitelist(html);
 43 		
 44 		if(this.urlValidationMode === 'relative') {
 45 			html = this.makeUrlsRelative(html);
 46 		} else if(this.urlValidationMode === 'host_relative') {
 47 			html = this.makeUrlsHostRelative(html);
 48 		} else if(this.urlValidationMode === 'absolute') {
 49 			// Trident always use absolute URL so we don't need to do anything.
 50 			//
 51 			// html = this.makeUrlsAbsolute(html);
 52 		}
 53 		
 54 		return html;
 55 	},
 56 	
 57 	lowerTagNamesAndUniformizeQuotation: function(html) {
 58 		this.pAttrQuotation1 = xq.compilePattern("\\s(\\w+?)=\\s+\"([^\"]+)\"", "mg");
 59 		this.pAttrQuotation2 = xq.compilePattern("\\s(\\w+?)=([^ \"]+)", "mg");
 60 		this.pAttrQuotation3 = xq.compilePattern("\\sNAME=\"(\\w+?)\" VALUE=\"(\\w+?)\"", "mg");
 61 
 62 		// Uniformize quotation, turn tag names and attribute names into lower case
 63 		html = html.replace(/<(\/?)(\w+)([^>]*?)>/img, function(str, closingMark, tagName, attrs) {
 64 			return "<" + closingMark + tagName.toLowerCase() + this.correctHtmlAttrQuotation(attrs) + ">";
 65 		}.bind(this));
 66 		
 67 		return html;
 68 	},
 69 	
 70 	correctHtmlAttrQuotation: function(html) {
 71 		html = html.replace(this.pAttrQuotation1, function (str, name, value) {return " " + name.toLowerCase() + '=' + '"' + value + '"'});
 72 		html = html.replace(this.pAttrQuotation2, function (str, name, value) {return " " + name.toLowerCase() + '=' + '"' + value + '"'});
 73 		html = html.replace(this.pAttrQuotation3, function (str, name, value) {return " name=\"" + name + "\" value=\"" + value + "\""});
 74 		return html;
 75 	}
 76 });
