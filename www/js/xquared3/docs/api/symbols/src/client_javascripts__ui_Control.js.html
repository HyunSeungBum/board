  7 xq.ui.FormDialog = xq.Class(/** @lends xq.ui.FormDialog.prototype */ {
  8 	/**
  9      * Displays given HTML form as a dialog.
 10      * 
 11      * @constructs
 12      * @param {xq.Editor} xed Dialog owner.
 13      * @param {String} html HTML string which contains FORM.
 14      * @param {Function} [onLoadHandler] callback function to be called when the form is loaded.
 15      * @param {Function} [onCloseHandler] callback function to be called when the form is closed.
 16 	 */
 17 	initialize: function(xed, html, onLoadHandler, onCloseHandler) {
 18 		xq.addToFinalizeQueue(this);
 19 		
 20 		this.xed = xed;
 21 		this.html = html;
 22 		this.onLoadHandler = onLoadHandler || function() {};
 23 		this.onCloseHandler = onCloseHandler || function() {};
 24 		this.form = null;
 25 	},
 26 	
 27 	/**
 28 	 * Shows dialog
 29 	 *
 30 	 * @param {Object} [options] collection of options
 31 	 */
 32 	show: function(options) {
 33 		options = options || {};
 34 		options.position = options.position || 'centerOfWindow';
 35 		options.mode = options.mode || 'modal';
 36 		options.cancelOnEsc = options.cancelOnEsc || true;
 37 		
 38 		if( typeof options.dialogId === 'undefined' )
 39 		{
 40 			this._createDialog(options);
 41 		}
 42 		else
 43 		{
 44 			if(!xq.$(options.dialogId))
 45 				this._createDialog(options);
 46 			else
 47 				xq.$(options.dialogId).style.display = '';				
 48 		}
 49 	},
 50 	
 51 	_createDialog: function(options)
 52 	{
 53 		var self = this;
 54 		
 55 		// create and append container
 56 		var container = document.createElement('DIV');
 57 		container.style.display = 'none';
 58 		document.body.appendChild(container);
 59 		
 60 		// initialize form
 61 		container.innerHTML = this.html;
 62 		this.form = container.getElementsByTagName('FORM')[0];
 63 		
 64 		var submitButtons = xq.getElementsByClassName(this.form, 'submit');
 65 		for (var i = 0; i < submitButtons.length; i++) {
 66 			submitButtons[i].onclick = function() {
 67 				self.onCloseHandler(xq.serializeForm(this.form));
 68 				if(!options.notSelfClose)
 69 				{
 70 					self.close();
 71 				}
 72 				return false;
 73 			}.bind(this);
 74 		}
 75 		
 76 		var cancelButtons = xq.getElementsByClassName(this.form, 'cancel');
 77 		for (var j = 0; j < cancelButtons.length; j++) {
 78 			cancelButtons[j].onclick = function() {
 79 				self.onCloseHandler();
 80 				self.close();
 81 			};
 82 		}
 83 		
 84 		if(options.mode === 'modal') {
 85 			this.dimmed = document.createElement('DIV');
 86 			this.dimmed.style.position = 'absolute';
 87 			this.dimmed.style.backgroundColor = 'black';
 88 			this.dimmed.style.opacity = 0.5;
 89 			this.dimmed.style.filter = 'alpha(opacity=50)';
 90 			this.dimmed.style.zIndex=902;
 91 			this.dimmed.style.top='0px';
 92 			this.dimmed.style.left='0px';
 93 			this.xed.outmostWrapper.appendChild(this.dimmed);
 94 			
 95 			this.resizeDimmedDiv = function(e) {
 96 				this.dimmed.style.display='none';
 97 				this.dimmed.style.width= this.xed.outmostWrapper.offsetWidth+'px';
 98 				this.dimmed.style.height= this.xed.outmostWrapper.offsetHeight+'px';
 99 				this.dimmed.style.display='block';
100 			}.bind(this);
101 			
102 			xq.observe(window, 'resize', this.resizeDimmedDiv);
103 			
104 			this.resizeDimmedDiv();
105 		}
106 		
107 		// append dialog
108 		this.xed.toolbar.dialogContainer.appendChild(this.form);
109 		container.parentNode.removeChild(container);
110 		
111 		// place dialog to center of window
112 		this.setPosition(options.position);
113 		
114 		// give focus
115 		var elementToFocus = xq.getElementsByClassName(this.form, 'initialFocus');
116 		if(elementToFocus.length > 0) {
117 			setTimeout(function(){ // @WORKAROUND: required to avoid Windows IE focus bug.
118 				elementToFocus[0].focus();
119 			},0);
120 		}
121 		
122 		// handle cancelOnEsc option
123 		if(options.cancelOnEsc) {
124 			xq.observe(this.form, 'keydown', function(e) {
125 				if(e.keyCode === 27) {
126 					this.onCloseHandler();
127 					this.close();
128 				}
129 			}.bind(this));
130 		}
131 		
132 		this.onLoadHandler(this);
133 	},
134 	/**
135 	 * Closes dialog
136 	 */
137 	close: function() {
138 		if(this.form.parentNode) this.form.parentNode.removeChild(this.form);
139 		
140 		if(this.dimmed) {
141 			this.dimmed.parentNode.removeChild(this.dimmed);
142 			this.dimmed = null;
143 			xq.stopObserving(window, 'resize', this.resizeDimmedDiv);
144 			this.resizeDimmedDiv = null;
145 		}
146 	},
147 	
148 	/**
149 	 * Sets position of dialog
150 	 *
151 	 * @param {String} target "centerOfWindow" or "centerOfEditor"
152 	 */
153 	setPosition: function(target) {
154 		var targetElement = null;
155 		var left = 0;
156 		var top = 0;
157 		if(target === 'centerOfWindow') {
158 			targetElement = document.documentElement || document.body;
159 			left += targetElement.scrollLeft;
160 			top += targetElement.scrollTop;
161 		} else if(target === 'centerOfEditor') {
162 			targetElement = this.xed.getCurrentEditMode() == 'wysiwyg' ? this.xed.wysiwygEditorDiv : this.xed.sourceEditorDiv;
163 			/*
164 			var o = targetElement;
165 			do {
166 				left += o.offsetLeft;
167 				top += o.offsetTop;
168 			} while(o = o.offsetParent)
169 			*/ 
170 		} else if(target === 'nearbyCaret') {
171 			throw "Not implemented yet";
172 		} else {
173 			throw "Invalid argument: " + target;
174 		}
175 		
176 		var targetWidth = targetElement.clientWidth;
177 		var targetHeight = targetElement.clientHeight;
178 		var dialogWidth = this.form.clientWidth;
179 		var dialogHeight = this.form.clientHeight;
180 		
181 		left += parseInt((targetWidth - dialogWidth) / 2);
182 		top += parseInt((targetHeight - dialogHeight) / 2);
183 		
184 		this.form.style.left = Math.max(left, 0) + "px";
185 		this.form.style.top = Math.max(top, 0) + "px";
186 	}
187 })

191 xq.ui.QuickSearchDialog = xq.Class(/** @lends xq.ui.QuickSearchDialog.prototype */ {
192 	/**
193      * Displays quick search dialog
194      * 
195      * @constructs
196      * @param {xq.Editor} xed Dialog owner.
197      * @param {Object} param Parameters.
198 	 */
199 	initialize: function(xed, param) {
200 		xq.addToFinalizeQueue(this);
201 		this.xed = xed;
202 		
203 		this.rdom = xq.rdom.Base.createInstance();
204 		
205 		this.param = param;
206 		if(!this.param.renderItem) this.param.renderItem = function(item) {
207 			return this.rdom.getInnerText(item);
208 		}.bind(this);
209 		
210 		this.container = null;
211 	},
212 	
213 	getQuery: function() {
214 		if(!this.container) return "";
215 		return this._getInputField().value;
216 	},
217 	
218 	onSubmit: function(e) {
219 		if(this.matchCount() > 0) {
220 			this.param.onSelect(this.xed, this.list[this._getSelectedIndex()]);
221 		}
222 		
223 		this.close();
224 		xq.stopEvent(e);
225 		return false;
226 	},
227 	
228 	onCancel: function(e) {
229 		if(this.param.onCancel) this.param.onCancel(this.xed);
230 		this.close();
231 	},
232 	
233 	onBlur: function(e) {
234 		// @WORKAROUND: Ugly
235 		setTimeout(function() {this.onCancel(e)}.bind(this), 400);
236 	},
237 	
238 	onKey: function(e) {
239 		var esc = new xq.Shortcut("ESC");
240 		var enter = new xq.Shortcut("ENTER");
241 		var up = new xq.Shortcut("UP");
242 		var down = new xq.Shortcut("DOWN");
243 		
244 		if(esc.matches(e)) {
245 			this.onCancel(e);
246 		} else if(enter.matches(e)) {
247 			this.onSubmit(e);
248 		} else if(up.matches(e)) {
249 			this._moveSelectionUp();
250 		} else if(down.matches(e)) {
251 			this._moveSelectionDown();
252 		} else {
253 			this.updateList();
254 		}
255 	},
256 	
257 	onClick: function(e) {
258 		var target = e.srcElement || e.target;
259 		if(target.nodeName === "LI") {
260 			
261 			var index = this._getIndexOfLI(target);
262 			this.param.onSelect(this.xed, this.list[index]);
263 		}
264 	},
265 	
266 	onList: function(list) {
267 		this.list = list;
268 		this.renderList(list);
269 	},
270 	
271 	updateList: function() {
272 		window.setTimeout(function() {
273 			this.param.listProvider(this.getQuery(), this.xed, this.onList.bind(this));
274 		}.bind(this), 0);
275 	},
276 	
277 	renderList: function(list) 
278 	{
279 		var ol = this._getListContainer();
280 		ol.innerHTML = "";
281 		
282 		for(var i = 0; i < list.length; i++) {
283 			var li = this.rdom.createElement('LI');
284 			li.innerHTML = this.param.renderItem(list[i]);
285 			ol.appendChild(li);
286 		}
287 		
288 		if(ol.hasChildNodes()) {
289 			ol.firstChild.className = "selected";
290 		}
291 	},
292 	
293 	show: function() {
294 		if(!this.container) this.container = this._create();
295 		
296 		var dialog = this.rdom.insertNodeAt(this.container, this.rdom.getRoot(), "end");
297 		this.setPosition('centerOfEditor');
298 		this.updateList();
299 		this.focus();
300 	},
301 	
302 	close: function() {
303 		this.rdom.deleteNode(this.container);
304 	},
305 	
306 	focus: function() {
307 		this._getInputField().focus();
308 	},
309 	
310 	setPosition: function(target) {
311 		var targetElement = null;
312 		var left = 0;
313 		var top = 0;
314 		
315 		if(target === 'centerOfWindow') {
316 			left += targetElement.scrollLeft;
317 			top += targetElement.scrollTop;
318 			targetElement = document.documentElement;
319 		} else if(target === 'centerOfEditor') {
320 			targetElement = this.xed.getCurrentEditMode() == 'wysiwyg' ? this.xed.wysiwygEditorDiv : this.xed.sourceEditorDiv;
321 			var o = targetElement;
322 			do {
323 				left += o.offsetLeft;
324 				top += o.offsetTop;
325 			} while(o = o.offsetParent)
326 		} else if(target === 'nearbyCaret') {
327 			throw "Not implemented yet";
328 		} else {
329 			throw "Invalid argument: " + target;
330 		}
331 		
332 		var targetWidth = targetElement.clientWidth;
333 		var targetHeight = targetElement.clientHeight;
334 		var dialogWidth = this.container.clientWidth;
335 		var dialogHeight = this.container.clientHeight;
336 		
337 		left += parseInt((targetWidth - dialogWidth) / 2);
338 		top += parseInt((targetHeight - dialogHeight) / 2);
339 		
340 		this.container.style.left = left + "px";
341 		this.container.style.top = top + "px";
342 	},
343 	
344 	matchCount: function() {
345 		return this.list ? this.list.length : 0;
346 	},
347 	
348 	_create: function() {
349 		// make container
350 		var container = this.rdom.createElement("DIV");
351 		container.className = "xqQuickSearch";
352 		
353 		// make title
354 		if(this.param.title) {
355 			var title = this.rdom.createElement("H1");
356 			title.innerHTML = this.param.title;
357 			container.appendChild(title);
358 		}
359 		
360 		// make input field
361 		var inputWrapper = this.rdom.createElement("DIV");
362 		inputWrapper.className = "input";
363 		var form = this.rdom.createElement("FORM");
364 		var input = this.rdom.createElement("INPUT");
365 		input.type = "text";
366 		input.value = "";
367     	form.appendChild(input);
368 		inputWrapper.appendChild(form);
369 		container.appendChild(inputWrapper);
370 		
371 		// make list
372 		var list = this.rdom.createElement("OL");
373 
374 	    xq.observe(input, 'blur', this.onBlur.bindAsEventListener(this));
375     	xq.observe(input, 'keypress', this.onKey.bindAsEventListener(this));
376     	xq.observe(list, 'click', this.onClick.bindAsEventListener(this), true);
377     	xq.observe(form, 'submit', this.onSubmit.bindAsEventListener(this));
378     	xq.observe(form, 'reset', this.onCancel.bindAsEventListener(this));
379 
380 		container.appendChild(list);
381 		return container;
382 	},
383 	
384 	_getInputField: function() {
385 		return this.container.getElementsByTagName('INPUT')[0];
386 	},
387 	
388 	_getListContainer: function() {
389 		return this.container.getElementsByTagName('OL')[0];
390 	},
391 	
392 	_getSelectedIndex: function() {
393 		var ol = this._getListContainer();
394 		for(var i = 0; i < ol.childNodes.length; i++) {
395 			if(ol.childNodes[i].className === 'selected') return i;
396 		}
397 	},
398 	
399 	_getIndexOfLI: function(li) {
400 		var ol = this._getListContainer();
401 		for(var i = 0; i < ol.childNodes.length; i++) {
402 			if(ol.childNodes[i] === li) return i;
403 		}
404 	},
405 	
406 	_moveSelectionUp: function() {
407 		var count = this.matchCount();
408 		if(count === 0) return;
409 		var index = this._getSelectedIndex();
410 		var ol = this._getListContainer();
411 		ol.childNodes[index].className = "";
412 		
413 		index--;
414 		if(index < 0) index = count - 1;
415 		
416 		ol.childNodes[index].className = "selected";
417 	},
418 	
419 	_moveSelectionDown: function() {
420 		var count = this.matchCount();
421 		if(count === 0) return;
422 		var index = this._getSelectedIndex();
423 		var ol = this._getListContainer();
424 		ol.childNodes[index].className = "";
425 		
426 		index++;
427 		if(index >= count) index = 0;
428 		
429 		ol.childNodes[index].className = "selected";
430 	}
