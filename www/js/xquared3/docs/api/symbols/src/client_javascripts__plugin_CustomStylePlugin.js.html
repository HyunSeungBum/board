1 /**
  2  * @requires Xquared.js
  3  * @requires Browser.js
  4  * @requires Editor.js
  5  * @requires plugin/Base.js
  6  * @requires ui/Control.js
  7  */
  8 xq.plugin.CustomStylePlugin = xq.Class(xq.plugin.Base,
  9 	/**
 10 	 * @name xq.plugin.CustomStylePlugin
 11 	 * @lends xq.plugin.CustomStylePlugin.prototype
 12 	 * @extends xq.plugin.Base
 13 	 * @constructor
 14 	 */
 15 	{
 16 	onAfterLoad: function(xed) {
 17 		xed.config.defaultToolbarButtonGroups.block.push(
 18 			{className:"lineHeight", title:"Line Height", list: [
 19 				{html:"50%25", style: {marginBottom: "3px"}, handler:"xed.handleLineHeight('50%')"},
 20 				{html:"80%25", style: {marginBottom: "3px"}, handler:"xed.handleLineHeight('80%')"},
 21 				{html:"100%25", style: {marginBottom: "3px"}, handler:"xed.handleLineHeight('100%')"},
 22 				{html:"120%25", style: {marginBottom: "3px"}, handler:"xed.handleLineHeight('120%')"},
 23 				{html:"150%25", style: {marginBottom: "3px"}, handler:"xed.handleLineHeight('150%')"},
 24 				{html:"180%25", style: {marginBottom: "3px"}, handler:"xed.handleLineHeight('180%')"},
 25 				{html:"200%25", style: {marginBottom: "3px"}, handler:"xed.handleLineHeight('200%')"}
 26 			]}
 27 		)
 28 			
 29 		xed.handleLineHeight = function(value){
 30 			if(xed.rdom.hasSelection(true)) {
 31 				var blocks = xed.rdom.getBlockElementsAtSelectionEdge(true, true);
 32 				if(blocks.first() !== blocks.last()) {
 33 					xed.rdom.applyLineHeights(blocks.first(), blocks.last(), value);
 34 					
 35 					var historyAdded = xed.editHistory.onCommand();
 36 					xed._fireOnCurrentContentChanged(xed);
 37 					
 38 					return true;
 39 				}
 40 			}
 41 			
 42 			var affected = xed.rdom.applyLineHeight(value);
 43 			
 44 			if(affected && !xed.rdom.tree.isAtomic(xed.rdom.getCurrentElement())) {
 45 				xed.rdom.placeCaretAtStartOf(affected);
 46 				
 47 				var historyAdded = xed.editHistory.onCommand();
 48 				xed._fireOnCurrentContentChanged(this);
 49 			}
 50 			
 51 			return true;
 52 		}
 53 		
 54 		xed.rdom.applyLineHeight = function(value, element){
 55 			element = element || xed.rdom.getCurrentBlockElement();
 56 			
 57 			var root = xed.rdom.getRoot();
 58 			if(!element || element === root) return null;
 59 		
 60 			element.style.lineHeight = value;
 61 			
 62 			return element;
 63 		}
 64 		
 65 		xed.rdom.applyLineHeights = function(from, to, value){
 66 			var blocks = xed.rdom.getBlockElementsBetween(from, to);
 67 			for (var i=0; i < blocks.length; i++) {
 68 				if (xed.rdom.tree._blockContainerTags.indexOf(blocks[i].nodeName) === -1 && xed.rdom.tree._blockTags.indexOf(blocks[i].nodeName) !== -1) {
 69 					xed.rdom.applyLineHeight(value, blocks[i]);
 70 				}
 71 			}
 72 		}
 73 	}
 74 });
