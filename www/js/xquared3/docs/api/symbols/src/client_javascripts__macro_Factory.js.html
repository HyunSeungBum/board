1 /**
  2  * @requires XQuared.js
  3  * @requires macro/Base.js
  4  */
  5 xq.macro.Factory = xq.Class(/** @lends xq.macro.Factory.prototype */{
  6 	/**
  7      * @constructs
  8      *
  9      * @param {String} URL to place holder image.
 10 	 */
 11 	initialize: function(placeHolderImgSrc) {
 12 		this.placeHolderImgSrc = placeHolderImgSrc;
 13 		this.macroClazzes = {};
 14 	},
 15 	/**
 16 	 * Registers new macro by ID.
 17 	 *
 18 	 * @param {String} id Macro id.
 19 	 */
 20 	register: function(id) {
 21 		var clazz = xq.macro[id + "Macro"];
 22 		if(!clazz) throw "Unknown macro id: [" + id + "]";
 23 		
 24 		this.macroClazzes[id] = clazz;
 25 	},
 26 	/**
 27 	 * Creates macro instance by given HTML fragment.
 28 	 *
 29 	 * @param {String} html HTML fragment.
 30 	 * @returns {xq.macro.Base} Macro instance or null if recognization of the HTML fragment fails.
 31 	 */
 32 	createMacroFromHtml: function(html) {
 33 		for(var id in this.macroClazzes) {
 34 			var clazz = this.macroClazzes[id];
 35 			if(clazz.recognize(html)) return new clazz(id, html, this.placeHolderImgSrc);
 36 		}
 37 		return null;
 38 	},
 39 	/**
 40 	 * Creates macro instance by given macro definition.
 41 	 *
 42 	 * @param {Object} def Macro definition.
 43 	 * @returns {xq.macro.Base} Macro instance
 44 	 * @throws If macro not found by def[id].
 45 	 */
 46 	createMacroFromDefinition: function(def) {
 47 		var clazz = this.macroClazzes[def.id];
 48 		if(!clazz) return null;
 49 		
 50 		return new clazz(def.id, def.params, this.placeHolderImgSrc);
 51 	}
 52 })
